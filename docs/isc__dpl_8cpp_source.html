90{
91

93 return ISCDPL_E_OPVERLAPED_OPERATION;
94 }
95

97 int ret = isc_main_control_->Initialize(ipc_dpl_configuratio);
98 if (ret != DPC_E_OK) {
99 return ret;
100 }
101
102 return DPC_E_OK;
