236int IscDataprocResultdataRingBuffer::GetPutBuffer(BufferData** buffer_data, const ULONGLONG time)
237{
238 if (buffer_data_ == nullptr) {
239 return -1;
240 }
241
242 EnterCriticalSection(&flag_critical_);
243
244 int local_write_inex = write_inex_;
245
246 if (buffer_data_[local_write_inex].state == 3) {
247 // in use for Get...
248 LeaveCriticalSection(&flag_critical_);
249 return -1;
250 }
251
252 if (!allow_overwrite_) {
253 if (buffer_data_[local_write_inex].state != 0) {
254 LeaveCriticalSection(&flag_critical_);
255 return -1;
256 }
257 }
258
259 *buffer_data = &buffer_data_[local_write_inex];
260
261 buffer_data_[local_write_inex].time = time; // GetTickCount();
262 buffer_data_[local_write_inex].state = 1;
263 put_index_ = local_write_inex;
264
265 LeaveCriticalSection(&flag_critical_);
266
267 return put_index_;
268}

279int IscDataprocResultdataRingBuffer::DonePutBuffer(const int index, const int status)
280{
281 // status 0:This data is invalid 1:This data is valid
282
283 if (buffer_data_ == nullptr) {
284 return -1;
285 }
286
287 if (index < 0 || index >= buffer_count_) {
288 return -1;
289 }
290
291 if (index != put_index_) {
292 // Get and Done must correspond one-to-one on the same Thread
293 return -1;
294 }
295
296 EnterCriticalSection(&flag_critical_);
297
298 if (buffer_data_[index].state != 1) {
299 // error, this case should not exist
300 __debugbreak();
301 LeaveCriticalSection(&flag_critical_);
302 return -1;
303 }
304
305 if (status == 0) {
306 // it change 1 -> 0 (not use)
307 buffer_data_[index].state = 0;
308 }
309 else {
310 buffer_data_[index].state = 2;
311
312 if (last_mode_) {
313 read_index_ = index;
314 }
315
316 write_inex_++;
317 if (write_inex_ >= buffer_count_) {
318 write_inex_ = 0;
319 }
320 }
321
322 LeaveCriticalSection(&flag_critical_);
323
324 return 0;
325}

337int IscDataprocResultdataRingBuffer::GetGetBuffer(BufferData** buffer_data, ULONGLONG* time_get)
338{
339 if (buffer_data_ == nullptr) {
340 return -1;
341 }
342
343 EnterCriticalSection(&flag_critical_);
344 int local_read_index = read_index_;
345
346 if (buffer_data_[local_read_index].state != 2) {
347 LeaveCriticalSection(&flag_critical_);
348 return -1;
349 }
350
351 *buffer_data = &buffer_data_[local_read_index];
352
353 *time_get = buffer_data_[local_read_index].time;
354 buffer_data_[local_read_index].state = 3;
355 geted_inedx_ = local_read_index;
356
357 if (last_mode_) {
358 if (!allow_overwrite_) {
359 // clear previous
360 int s = local_read_index - 1;
361 int e = local_read_index + 1;
362 if (s == -1) {
363 s = buffer_count_ - 1;
364 }
365 if (e >= buffer_count_) {
366 e = 0;
367 }
368 for (int i = s; i != e;) {
369 if (buffer_data_[i].state != 2) {
370 break;
371 }
372 buffer_data_[i].state = 0;
373
374 i--;
375 if (i == 0) {
376 i = buffer_count_ - 1;
377 }
378 }
379 }
380 }
381 else {
382 read_index_++;
383 if (read_index_ >= buffer_count_) {
384 read_index_ = 0;
385 }
386 }
387
388 LeaveCriticalSection(&flag_critical_);
389
390 return geted_inedx_;
391}
