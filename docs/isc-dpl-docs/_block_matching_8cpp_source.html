450void BlockMatching::executeMatching(int imghgt, int imgwdt, int depth,
451 unsigned char *pimgref, unsigned char *pimgcmp,
452 float * pblkdsp, int *pblkcrst)
453{
454
455 // 視差ブロックの高さ
456 int stphgt = disparityBlockHeight;
457 // 視差ブロックの幅
458 int stpwdt = disparityBlockWidth;
459 // マッチングブロックの高さ
460 int blkhgt = matchingBlockHeight;
461 // マッチングブロックの幅
462 int blkwdt = matchingBlockWidth;
463
464 // 出力視差ブロック画像の高さ
465 // * 小数切り捨て
466 int imghgtblk = imghgt / stphgt;
467 // 出力視差ブロック画像の幅
468 // * 小数切り捨て
469 int imgwdtblk = imgwdt / stpwdt;
470
471 // コントラスト閾値
472 int crstthr = crstThr;
473
474 // コントラストオフセット
475 int crstofs = 0;
476
477 if (imgwdt == IMG_WIDTH_VM) {
478 crstofs = (int)CONTRAST_OFFSET_VM;
479 }
480 else if (imgwdt == IMG_WIDTH_XC) {
481 crstofs = (int)CONTRAST_OFFSET_XC;
482 }
483
484 // ブロック輝度最大値
485 int bgtmax = BLOCK_BRIGHTNESS_MAX;
486
487 // バックマッチング
488 float *pblkbkdsp = NULL;
489 shadeWidth = depth;
490
491 if (enableBackMatching == 1) {
492 memset(bk_block_dsp, 0, imghgt * imgwdt * sizeof(float));
493 pblkbkdsp = bk_block_dsp;
494 shadeWidth = 0;
495 }
496
497 // 視差を取得する
498 getMatchingDisparity(imghgt, imgwdt, depth, crstthr, crstofs, bgtmax,
499 stphgt, stpwdt, blkhgt, blkwdt, imghgtblk, imgwdtblk,
500 pimgref, pimgcmp, pblkdsp, pblkbkdsp, pblkcrst);
501
502 if (enableBackMatching == 1) {
503 blendBothMatchingDisparity(imghgt, imgwdt, imghgtblk, imgwdtblk,
504 backMatchingEvaluationWidth, backMatchingEvaluationRange, backMatchingValidRatio, backMatchingZeroRatio,
505 pblkdsp, pblkbkdsp);
506 }

650void BlockMatching::getDisparityBySSD(int x, int y, int imghgt, int imgwdt, int depth,
651 int crstthr, int crstofs, int bgtmax,
652 int stphgt, int stpwdt, int blkhgt, int blkwdt, int imghgtblk, int imgwdtblk,
653 unsigned char *pimgref, unsigned char *pimgcmp, float * pblkdsp, int *pblkcrst)
654{
655
656 // 視差画像の幅
657 int imgwdtdsp = imgwdt - depth;
658 int jpx = y;
659 int ipx = x;
660 if (ipx >= imgwdtdsp || jpx % stphgt != 0 || ipx % stpwdt != 0) {
661 return;
662 }
663
664 // jblk : マッチングブロックのyインデックス
665 // iblk : マッチングブロックのxインデックス
666 int jblk = jpx / stphgt;
667 int iblk = ipx / stpwdt;
668
669 // SSDを保存する配列
670 // 配列サイズはマッチング探索幅
671 float ssd[ISC_IMG_DEPTH_MAX];
672 // SSD最大値を設定
673 float misum = FLT_MAX;
674
675 // 視差値の初期値を設定する
676 int disp = 0;
677
678 // マッチング評価関数
679 // 左右画像の同じ位置iにある輝度の差の二乗を求め，それを領域全体に渡って足し合わせる
680 // SSD = Σ(((右画像の輝度ij - 右画像ブロック輝度平均) - (左画像の輝度ij - 左画像ブロック輝度平均))^2)
681 // 右画像を1画素ずつ探索範囲DEPTHまで右へ動かしてSSDを計算する
682 // 一番小さいSSDのとき，最も類似している
683 // その探索位置を視差値（整数部）とする
684 // 前後のSSDからサブピクセル値（少数部）を算出する
685 int jpxe = jpx + blkhgt;
686 int ipxe = ipx + blkwdt;
687
688 int sumr = 0; // Σ基準画像の輝度ij
689 int sumrr = 0; // Σ(基準画像の輝度ij^2)
690
691 int Lmin = 255; // ブロック内輝度最小値（初期値256階調最大輝度値）
692 int Lmax = 0; // ブロック内輝度最大値（初期値256階調最小輝度値）
693
694 int blkcnt = blkhgt * blkwdt;
695
696 // 右画像ブロック輝度を評価する
697 for (int j = jpx; j < jpxe; j++) {
698 for (int i = ipx; i < ipxe; i++) {
699 int rfx = pimgref[j * imgwdt + i];
700
701 sumr += rfx;
702 sumrr += rfx * rfx;
703
704 if (Lmin > rfx) {
705 Lmin = rfx;
706 }
707 else if (Lmax < rfx) {
708 Lmax = rfx;
709 }
710
711 }
712 }
713
714 // コントラストを求める
715 int crst = 0;
716 if (Lmax >= bgtmax) {
717 crst = ((Lmax - Lmin) * 1000 - crstofs) * blkcnt / sumr;
718 }
719
720 // コントラストが閾値未満の場合は視差値ゼロにする
721 if (crst < crstthr) {
722 // 視差値
723 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
724 // コントラスト
725 pblkcrst[jblk * imgwdtblk + iblk] = 0;
726 return;
727 }
728
729 for (int k = 0; k < depth; k++) {
730 float sumsq = 0.0;
731
732 int sumc = 0; // Σ比較画像の輝度ij
733 int sumcc = 0; // Σ(比較画像の輝度ij^2)
734 int sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
735
736 for (int j = jpx; j < jpxe; j++) {
737 for (int i = ipx; i < ipxe; i++) {
738 int rfx = pimgref[j * imgwdt + i];
739 int cpx = pimgcmp[j * imgwdt + i + k];
740
741 sumc += cpx;
742 sumcc += cpx * cpx;
743 sumrc += rfx * cpx;
744
745 }
746 }
747 sumsq = sumrr + sumcc - 2 * sumrc - (float)(sumr * sumr + sumc * sumc - 2 * sumr * sumc) / blkcnt;
748
749 // 右画像を1画素ずつ右へ動かしてSSDを計算する
750 // 一番小さいSSDのとき，最も類似している
751 ssd[k] = sumsq;
752 if (sumsq < misum) {
753 misum = sumsq;
754 disp = k;
755 }
756 }
757
758 // 視差値が探索開始スキップ幅：1未満の場合はサブピクセルを算出しない
759 // 視差値が探索幅の上限に達していた場合はサブピクセルを算出しない
760 // 視差値が1未満の場合は視差値ゼロにする
761 // 視差値が探索幅の上限に達していた場合は視差値ゼロにする
762 if (disp < 1 || disp >= (depth - 1)) {
763 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
764 }
765 else {
766 // サブピクセル推定
767 // 折れ線近似
768 // Xsub = (S(-1) - S(1)) / (2 x S(-1) - 2 x S(0)) S(-1) > S(1)
769 // Xsub = (S(1) - S(-1)) / (2 x S(1) - 2 x S(0)) S(-1) ≦ S(1)
770 // 放物線近似
771 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
772 // 折れ線+放物線近似
773 // Xsub = (S(-1) - S(1)) / (S(-1) - S(0) - S(-1) + S(2))
774 // Xsub = (S(1) - S(-1)) / (S(-2) - S(-1) - S(0) + S(1))
775 float sub;
776 float ssdprv = ssd[disp - 1];
777 float ssdcnt = ssd[disp];
778 float ssdnxt = ssd[disp + 1];
779
780 // 放物線近似
781 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
782
783 pblkdsp[jblk * imgwdtblk + iblk] = disp + sub;
784
785 }
786
787 // コントラストを保存する
788 pblkcrst[jblk * imgwdtblk + iblk] = crst;
789
790}

861 float * pblkdsp, float * pblkbkdsp, int *pblkcrst)
862{
863
864 // 視差画像の幅
865 int imgwdtdsp = imgwdt - blkwdt;
866 int jpx = y;
867 int ipx = x;
868 if (ipx >= imgwdtdsp || jpx % stphgt != 0 || ipx % stpwdt != 0) {
869 return;
870 }
871
872 // jblk : マッチングブロックのyインデックス
873 // iblk : マッチングブロックのxインデックス
874 int jblk = jpx / stphgt;
875 int iblk = ipx / stpwdt;
876
877 // SSDを保存する配列
878 // 配列サイズはマッチング探索幅
879 float ssd[ISC_IMG_DEPTH_MAX];
880 float bk_ssd[ISC_IMG_DEPTH_MAX];
881 // SSD最大値を設定
882 float misum = FLT_MAX;
883 float bk_misum = FLT_MAX;
884
885 // 視差値の初期値を設定する
886 int disp = 0;
887 int bk_disp = 0;
888
889 // マッチング評価関数
890 // 左右画像の同じ位置iにある輝度の差の二乗を求め，それを領域全体に渡って足し合わせる
891 // SSD = Σ(((右画像の輝度ij - 右画像ブロック輝度平均) - (左画像の輝度ij - 左画像ブロック輝度平均))^2)
892 // 右画像を1画素ずつ探索範囲DEPTHまで右へ動かしてSSDを計算する
893 // 一番小さいSSDのとき，最も類似している
894 // その探索位置を視差値（整数部）とする
895 // 前後のSSDからサブピクセル値（少数部）を算出する
896 int jpxe = jpx + blkhgt;
897 int ipxe = ipx + blkwdt;
898
899 // フォアマッチング
900 int sumr = 0; // Σ基準画像の輝度ij
901 int sumrr = 0; // Σ(基準画像の輝度ij^2)
902 int Lmin = 255; // ブロック内輝度最小値（初期値256階調最大輝度値）
903 int Lmax = 0; // ブロック内輝度最大値（初期値256階調最小輝度値）
904
905 // バックマッチング
906 int bk_sumr = 0; // Σ基準画像の輝度ij
907 int bk_sumrr = 0; // Σ(基準画像の輝度ij^2)
908 int bk_Lmin = 255; // ブロック内輝度最小値（初期値256階調最大輝度値）
909 int bk_Lmax = 0; // ブロック内輝度最大値（初期値256階調最小輝度値）
910
911 int blkcnt = blkhgt * blkwdt;
912
913 // 基準画像ブロック輝度を評価する
914 for (int j = jpx; j < jpxe; j++) {
915 for (int i = ipx; i < ipxe; i++) {
916 int rfx = pimgref[j * imgwdt + i]; // 基準（右）カメラ
917 int cpx = pimgcmp[j * imgwdt + i]; // 比較（左）カメラ
918
919 // フォアマッチング
920 sumr += rfx;
921 sumrr += rfx * rfx;
922 if (Lmin > rfx) {
923 Lmin = rfx;
924 }
925 else if (Lmax < rfx) {
926 Lmax = rfx;
927 }
928
929 // バックマッチング
930 bk_sumr += cpx;
931 bk_sumrr += cpx * cpx;
932 if (cpx > bk_Lmax) {
933 bk_Lmax = cpx;
934 }
935 else if (cpx < bk_Lmin) {
936 bk_Lmin = cpx;
937 }
938
939 }
940 }
941
942 // コントラストを求める
943 // フォアマッチング
944 int crst = 0;
945 if (Lmax >= bgtmax) {
946 crst = ((Lmax - Lmin) * 1000 - crstofs) * blkcnt / sumr;
947 }
948 // バックマッチング
949 int bk_crst = 0;
950 if (bk_Lmax >= bgtmax) {
951 bk_crst = ((bk_Lmax - bk_Lmin) * 1000 - crstofs) * blkcnt / bk_sumr;
952 }
953
954 // 探索マージンを求める
955 // フォアマッチング
956 int fr_mrgn = imgwdt - (ipx + depth + blkwdt);
957 // バックマッチング
958 int bk_mrgn = ipx - depth;
959 // 探索幅を求める
960 // フォワード
961 int fr_depth = depth;
962 if (fr_mrgn < 0) {
963 fr_depth = depth + fr_mrgn + 1;
964 }
965 // バックワード探索幅
966 int bk_depth = depth;
967 if (bk_mrgn < 0) {
968 bk_depth = depth + bk_mrgn + 1;
969 }
970
971 for (int k = 0; k < depth; k++) {
972
973 // フォアマッチング
974 int sumc = 0; // Σ比較画像の輝度ij
975 int sumcc = 0; // Σ(比較画像の輝度ij^2)
976 int sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
977 // バックマッチング
978 int bk_sumc = 0; // Σ比較画像の輝度ij
979 int bk_sumcc = 0; // Σ(比較画像の輝度ij^2)
980 int bk_sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
981
982 for (int j = jpx; j < jpxe; j++) {
983 for (int i = ipx; i < ipxe; i++) {
984 // フォアマッチング
985 if (k < fr_depth) {
986 int rfx = pimgref[j * imgwdt + i];
987 int cpx = pimgcmp[j * imgwdt + i + k];
988 sumc += cpx;
989 sumcc += cpx * cpx;
990 sumrc += rfx * cpx;
991 }
992 // バックマッチング
993 if (k < bk_depth) {
994 int rfx = pimgcmp[j * imgwdt + i];
995 int cpx = pimgref[j * imgwdt + i - k];
996 bk_sumc += cpx;
997 bk_sumcc += cpx * cpx;
998 bk_sumrc += rfx * cpx;
999 }
1000 }
1001 }
1002 // フォアマッチング
1003 float sumsq = sumrr + sumcc - 2 * sumrc - (float)(sumr * sumr + sumc * sumc - 2 * sumr * sumc) / blkcnt;
1004 // バックマッチング
1005 float bk_sumsq = bk_sumrr + bk_sumcc - 2 * bk_sumrc
1006 - (float)(bk_sumr * bk_sumr + bk_sumc * bk_sumc - 2 * bk_sumr * bk_sumc) / blkcnt;
1007
1008 // 右画像を1画素ずつ右へ動かしてSSDを計算する
1009 // 一番小さいSSDのとき，最も類似している
1010 // フォアマッチング
1011 ssd[k] = sumsq;
1012 if (sumsq < misum) {
1013 misum = sumsq;
1014 disp = k;
1015 }
1016 // バックマッチング
1017 bk_ssd[k] = bk_sumsq;
1018 if (bk_sumsq < bk_misum) {
1019 bk_misum = bk_sumsq;
1020 bk_disp = k;
1021 }
1022 }
1023
1024 // 視差値が探索開始スキップ幅：1未満の場合はサブピクセルを算出しない
1025 // 視差値が探索幅の上限に達していた場合はサブピクセルを算出しない
1026 // 視差値が1未満の場合は視差値ゼロにする
1027 // 視差値が探索幅の上限に達していた場合は視差値ゼロにする
1028 // コントラストが閾値未満の場合は視差値ゼロにする
1029 float sub;
1030 float ssdprv;
1031 float ssdcnt;
1032 float ssdnxt;
1033 // フォアマッチング
1034 if (fr_depth < 3 || disp < 1 || disp >= (fr_depth - 1) || crst < crstthr) {
1035 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
1036 crst = 0;
1037 }
1038 else {
1039 // 放物線近似
1040 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
1041 ssdprv = ssd[disp - 1];
1042 ssdcnt = ssd[disp];
1043 ssdnxt = ssd[disp + 1];
1044 // 放物線近似
1045 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
1046 // 視差値を保存する
1047 pblkdsp[jblk * imgwdtblk + iblk] = disp + sub;
1048 }
1049 // バックマッチング
1050 if (bk_depth >= 3 && bk_disp >= 1 && bk_disp < (bk_depth - 1) && bk_crst >= crstthr) {
1051 // 放物線近似
1052 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
1053 ssdprv = bk_ssd[bk_disp - 1];
1054 ssdcnt = bk_ssd[bk_disp];
1055 ssdnxt = bk_ssd[bk_disp + 1];
1056 // 放物線近似
1057 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
1058 float bk_disp_sub = bk_disp + sub;
1059 // バックマッチングの結果を基準画像の座標へ展開
1060 // 視差ブロック番号
1061 int bk_iblk = (int)((ipx - bk_disp_sub) / stpwdt);
1062 // 視差値を保存する
1063 pblkbkdsp[jblk * imgwdtblk + bk_iblk] = bk_disp_sub;
1064 }
1065
1066 // コントラストを保存する
1067 pblkcrst[jblk * imgwdtblk + iblk] = crst;
1068
1069}

1127 // 指定未満の場合は視差値ゼロにする
