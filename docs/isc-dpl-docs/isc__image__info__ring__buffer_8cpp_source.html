261int IscImageInfoRingBuffer::GetPutBuffer(BufferData** buffer_data, const ULONGLONG time)
262{
263 if (buffer_data_ == nullptr) {
264 return -1;
265 }
266
267 EnterCriticalSection(&flag_critical_);
268
269 int local_write_inex = write_inex_;
270
271 if (buffer_data_[local_write_inex].state == 3) {
272 // in use for Get...
273 LeaveCriticalSection(&flag_critical_);
274 return -1;
275 }
276
277 if (!allow_overwrite_) {
278 if (buffer_data_[local_write_inex].state != 0) {
279 LeaveCriticalSection(&flag_critical_);
280 return -1;
281 }
282 }
283
284 *buffer_data = &buffer_data_[local_write_inex];
285
286 buffer_data_[local_write_inex].time = time; // GetTickCount();
287 buffer_data_[local_write_inex].state = 1;
288 put_index_ = local_write_inex;
289
290 LeaveCriticalSection(&flag_critical_);
291
292 return put_index_;
293}

304int IscImageInfoRingBuffer::DonePutBuffer(const int index, const int status)
305{
306 // status 0:This data is invalid 1:This data is valid
307
308 if (buffer_data_ == nullptr) {
309 return -1;
310 }
311
312 if (index < 0 || index >= buffer_count_) {
313 return -1;
314 }
315
316 if (index != put_index_) {
317 // Get and Done must correspond one-to-one on the same Thread
318 return -1;
319 }
320
321 EnterCriticalSection(&flag_critical_);
322
323 if (buffer_data_[index].state != 1) {
324 // error, this case should not exist
325 __debugbreak();
326 LeaveCriticalSection(&flag_critical_);
327 return -1;
328 }
329
330 if (status == 0) {
331 // it change 1 -> 0 (not use)
332 buffer_data_[index].state = 0;
333 }
334 else {
335 buffer_data_[index].state = 2;
336
337 if (last_mode_) {
338 read_index_ = index;
339 }
340
341 write_inex_++;
342 if (write_inex_ >= buffer_count_) {
343 write_inex_ = 0;
344 }
345 }
346
347 LeaveCriticalSection(&flag_critical_);
348
349 return 0;
350}

361int IscImageInfoRingBuffer::GetGetBuffer(BufferData** buffer_data, ULONGLONG* time_get)
362{
363 if (buffer_data_ == nullptr) {
364 return -1;
365 }
366
367 EnterCriticalSection(&flag_critical_);
368 int local_read_index = read_index_;
369
370 if (buffer_data_[local_read_index].state != 2) {
371 LeaveCriticalSection(&flag_critical_);
372 return -1;
373 }
374
375 *buffer_data = &buffer_data_[local_read_index];
376
377 *time_get = buffer_data_[local_read_index].time;
378 buffer_data_[local_read_index].state = 3;
379 geted_inedx_ = local_read_index;
380
381 if (last_mode_) {
382 if (!allow_overwrite_) {
383 // clear previous
384 int s = local_read_index - 1;
385 int e = local_read_index + 1;
386 if (s == -1) {
387 s = buffer_count_ - 1;
388 }
389 if (e >= buffer_count_) {
390 e = 0;
391 }
392 for (int i = s; i != e;) {
393 if (buffer_data_[i].state != 2) {
394 break;
395 }
396 buffer_data_[i].state = 0;
397
398 i--;
399 if (i == 0) {
400 i = buffer_count_ - 1;
401 }
402 }
403 }
404 }
405 else {
406 read_index_++;
407 if (read_index_ >= buffer_count_) {
408 read_index_ = 0;
409 }
410 }
411
412 LeaveCriticalSection(&flag_critical_);
413
414 return geted_inedx_;
415}
