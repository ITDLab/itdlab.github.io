106int IscStereoMatchingInterface::Initialize(IscDataProcModuleConfiguration* isc_data_proc_module_configuration)
107{
108 swprintf_s(isc_data_proc_module_configuration_.configuration_file_path, L"%s", isc_data_proc_module_configuration->configuration_file_path);
109 swprintf_s(isc_data_proc_module_configuration_.log_file_path, L"%s", isc_data_proc_module_configuration->log_file_path);
110
111 isc_data_proc_module_configuration_.log_level = isc_data_proc_module_configuration->log_level;
112 isc_data_proc_module_configuration_.isc_camera_model = isc_data_proc_module_configuration->isc_camera_model;
113 isc_data_proc_module_configuration_.max_image_width = isc_data_proc_module_configuration->max_image_width;
114 isc_data_proc_module_configuration_.max_image_height = isc_data_proc_module_configuration->max_image_height;
115
116 switch (isc_data_proc_module_configuration_.isc_camera_model) {
117 case IscCameraModel::kVM:
118 swprintf_s(parameter_file_name_, L"%s\\StereoMatcingParameter_VM.ini", isc_data_proc_module_configuration_.configuration_file_path);
119 break;
120
121 case IscCameraModel::kXC:
122 swprintf_s(parameter_file_name_, L"%s\\StereoMatcingParameter_XC.ini", isc_data_proc_module_configuration_.configuration_file_path);
123 break;
124
125 case IscCameraModel::k4K:
126 swprintf_s(parameter_file_name_, L"%s\\StereoMatcingParameter_4K.ini", isc_data_proc_module_configuration_.configuration_file_path);
127 break;
128
129 case IscCameraModel::k4KA:
130 swprintf_s(parameter_file_name_, L"%s\\StereoMatcingParameter_4KA.ini", isc_data_proc_module_configuration_.configuration_file_path);
131 break;
132
133 case IscCameraModel::k4KJ:
134 swprintf_s(parameter_file_name_, L"%s\\StereoMatcingParameter_4KJ.ini", isc_data_proc_module_configuration_.configuration_file_path);
135 break;
136
137 default:
138 swprintf_s(parameter_file_name_, L"%s\\StereoMatcingParameter.ini", isc_data_proc_module_configuration_.configuration_file_path);
139 break;
140 }
141
142 int ret = LoadParameterFromFile(parameter_file_name_, &stereo_matching_parameters_);
143 if (ret != DPC_E_OK) {
144 if (ret == DPCPROCESS_E_FILE_NOT_FOUND) {
145 // Since the file does not exist, it will be created by default and continue
146 SaveParameterToFile(parameter_file_name_, &stereo_matching_parameters_);
147 }
148 else {
149 return ret;
150 }
151 }
152
153 stereo_matching_parameters_.matching_parameter.imghgt = isc_data_proc_module_configuration_.max_image_height;
154 stereo_matching_parameters_.matching_parameter.imgwdt = isc_data_proc_module_configuration_.max_image_width;
155
156 ret = SetParameterToStereoMatchingModule(&stereo_matching_parameters_);
157 if (ret != DPC_E_OK) {
158 return ret;
159 }
160
161 // get work
162 size_t image_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
163 size_t depth_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
164
165 for (int i = 0; i < 4; i++) {
166 work_buffers_.buff_image[i].width = 0;
167 work_buffers_.buff_image[i].height = 0;
168 work_buffers_.buff_image[i].channel_count = 0;
169 work_buffers_.buff_image[i].image = new unsigned char[image_size];
170 memset(work_buffers_.buff_image[i].image, 0, image_size);
171
172 work_buffers_.buff_depth[i].width = 0;
173 work_buffers_.buff_depth[i].height = 0;
174 work_buffers_.buff_depth[i].image = new float[depth_size];
175 memset(work_buffers_.buff_depth[i].image, 0, depth_size * sizeof(float));
176 }
177
178 // initialize FrameDecoder
179 StereoMatching::initialize(isc_data_proc_module_configuration_.max_image_height, isc_data_proc_module_configuration_.max_image_width);
180 StereoMatching::createMatchingThread();
181
182 // Check if OpenCL is available. Enable it if it is available.
183 if (stereo_matching_parameters_.system_parameter.enabled_opencl_for_avedisp && cv::ocl::haveOpenCL()) {
184 // it can use openCL
185 //cv::String build_info_str = cv::getBuildInformation();
186 //OutputDebugStringA(build_info_str.c_str());
187 StereoMatching::setUseOpenCLForMatching(1);
188 }
189 else {
190 StereoMatching::setUseOpenCLForMatching(0);
191 }
192
193 return DPC_E_OK;
194}

381int IscStereoMatchingInterface::Terminate()
382{
383 StereoMatching::deleteMatchingThread();
384
385 // release work
386 for (int i = 0; i < 4; i++) {
387 work_buffers_.buff_image[i].width = 0;
388 work_buffers_.buff_image[i].height = 0;
389 work_buffers_.buff_image[i].channel_count = 0;
390 delete[] work_buffers_.buff_image[i].image;
391 work_buffers_.buff_image[i].image = nullptr;
392
393 work_buffers_.buff_depth[i].width = 0;
394 work_buffers_.buff_depth[i].height = 0;
395 delete[] work_buffers_.buff_depth[i].image;
396 work_buffers_.buff_depth[i].image = nullptr;
397 }
398
399 return DPC_E_OK;
400

712 unsigned char* compare_image = isc_image_Info->frame_data[fd_index].p2.image;

716 // (2) get disparity

719 int height = isc_image_Info->frame_data[fd_index].p1.height;
