134 class LogDataQue {
135 CRITICAL_SECTION buffer_critical_;
136 bool mode_latest_;
137
138 int write_index_, read_index_, put_index_, geted_index_, last_done_index_;
139
140 int max_head_count_;
141 int max_data_count_;
142 int max_list_count_;
143
144 struct Data {
145 int index;
146 int state; // 0:nothing 1:write 3:read
147
148 SYSTEMTIME st;
149 Level level;
150 wchar_t head[256];
151 wchar_t data[1024];
152 };
153
154 Data* data_list_;
155
156 public:
157 LogDataQue()
158 {
159 max_head_count_ = 256;
160 max_data_count_ = 1024;
161 max_list_count_ = 256;
162 write_index_ = read_index_ = put_index_ = geted_index_ = last_done_index_ = 0;
163 data_list_ = nullptr;
164
165 mode_latest_ = true;
166 InitializeCriticalSection(&buffer_critical_);
167 }
168
169 ~LogDataQue()
170 {
171 DeleteCriticalSection(&buffer_critical_);
172 }
173
174 bool open(int max_list_count, bool latest)
175 {
176 if (max_list_count < 1) {
177 max_list_count = 1;
178 }
179 max_list_count_ = max_list_count;
180 write_index_ = read_index_ = put_index_ = geted_index_ = last_done_index_ = 0;
181
182 data_list_ = new Data[max_list_count_];
183 for (int i = 0; i < max_list_count_; i++) {
184 data_list_[i].index = i;
185 data_list_[i].state = 0;
186 data_list_[i].level = Level::Information;
187
188 memset(data_list_[i].head, 0, max_head_count_);
189 memset(data_list_[i].data, 0, max_data_count_);
190 }
191
192 mode_latest_ = latest;
193
194 return true;
195 }
196
197 bool close(void)
198 {
199 delete[] data_list_;
200 data_list_ = nullptr;
201
202 return true;
203 }
204
205 int put(Level level, const wchar_t* head, const wchar_t* data)
206 {
207 if (data == nullptr) {
208 return -1;
209 }
210
211 if (data_list_[write_index_].state != 0) {
212 return -1;
213 }
214
215 swprintf_s(data_list_[write_index_].head, L"%s", head);
216 swprintf_s(data_list_[write_index_].data, L"%s", data);
217
218 GetLocalTime(&data_list_[write_index_].st);
219 data_list_[write_index_].level = level;
220 data_list_[write_index_].state = 1;
221
222 EnterCriticalSection(&buffer_critical_);
223 if (mode_latest_) {
224 read_index_ = write_index_;
225 }
226 LeaveCriticalSection(&buffer_critical_);
227
228 put_index_ = write_index_;
229
230 write_index_++;
231 if (write_index_ >= max_list_count_) {
232 write_index_ = 0;
233 }
234
235 return put_index_;
236 }
237
238 int get(SYSTEMTIME* st, Level* level, wchar_t* head, int max_head_count, wchar_t* data, int max_data_count)
239 {
240 if (data == nullptr) {
241 return -1;
242 }
243 if (data_list_[read_index_].state != 1) {
244 return -1;
245 }
246
247 EnterCriticalSection(&buffer_critical_);
248 int cur_index = read_index_;
249 LeaveCriticalSection(&buffer_critical_);
250
251 swprintf_s(head, max_head_count, L"%s", data_list_[cur_index].head);
252 swprintf_s(data, max_data_count, L"%s", data_list_[cur_index].data);
253
254 *level = data_list_[cur_index].level;
255 *st = data_list_[cur_index].st;
256
257 data_list_[cur_index].state = 3;
258 geted_index_ = cur_index;
259
260 EnterCriticalSection(&buffer_critical_);
261 if (!mode_latest_) {
262 read_index_++;
263 if (read_index_ >= max_list_count_) {
264 read_index_ = 0;
265 }
266 }
267 LeaveCriticalSection(&buffer_critical_);
268
269 return geted_index_;
270 }
271
272 void done(int index)
273 {
274 if (data_list_ == nullptr) {
275 return;
276 }
277 if (index < 0 || index >= max_list_count_) {
278 return;
279 }
280 data_list_[index].state = 0;
281
282 if (mode_latest_) {
283 int st = last_done_index_;
284 int ed = index;
285
286 if (ed < st) {
287 for (int i = st; i < max_list_count_; i++) {
288 data_list_[i].state = 0;
289 }
290 for (int i = 0; i < ed; i++) {
291 data_list_[i].state = 0;
292 }
293 }
294 else {
295 for (int i = st; i < ed; i++) {
296 data_list_[i].state = 0;
297 }
298 }
299
300 last_done_index_ = index;
301 }
302
303 return;
304 }
305
306 };
307 LogDataQue* log_data_que_;
309};
