260int IscDataprocResultdataRingBuffer::GetPutBuffer(BufferData** buffer_data, const ULONGLONG time)
261{
262 if (buffer_data_ == nullptr) {
263 return -1;
264 }
265
266 EnterCriticalSection(&flag_critical_);
267
268 int local_write_inex = write_inex_;
269
270 if (buffer_data_[local_write_inex].state == 3) {
271 // in use for Get...
272 LeaveCriticalSection(&flag_critical_);
273 return -1;
274 }
275
276 if (!allow_overwrite_) {
277 if (buffer_data_[local_write_inex].state != 0) {
278 LeaveCriticalSection(&flag_critical_);
279 return -1;
280 }
281 }
282
283 *buffer_data = &buffer_data_[local_write_inex];
284
285 buffer_data_[local_write_inex].time = time; // GetTickCount();
286 buffer_data_[local_write_inex].state = 1;
287 put_index_ = local_write_inex;
288
289 LeaveCriticalSection(&flag_critical_);
290
291 return put_index_;
292}

303int IscDataprocResultdataRingBuffer::DonePutBuffer(const int index, const int status)
304{
305 // status 0:This data is invalid 1:This data is valid
306
307 if (buffer_data_ == nullptr) {
308 return -1;
309 }
310
311 if (index < 0 || index >= buffer_count_) {
312 return -1;
313 }
314
315 if (index != put_index_) {
316 // Get and Done must correspond one-to-one on the same Thread
317 return -1;
318 }
319
320 EnterCriticalSection(&flag_critical_);
321
322 if (buffer_data_[index].state != 1) {
323 // error, this case should not exist
324 __debugbreak();
325 LeaveCriticalSection(&flag_critical_);
326 return -1;
327 }
328
329 if (status == 0) {
330 // it change 1 -> 0 (not use)
331 buffer_data_[index].state = 0;
332 }
333 else {
334 buffer_data_[index].state = 2;
335
336 if (last_mode_) {
337 read_index_ = index;
338 }
339
340 write_inex_++;
341 if (write_inex_ >= buffer_count_) {
342 write_inex_ = 0;
343 }
344 }
345
346 LeaveCriticalSection(&flag_critical_);
347
348 return 0;
349}

361int IscDataprocResultdataRingBuffer::GetGetBuffer(BufferData** buffer_data, ULONGLONG* time_get)
362{
363 if (buffer_data_ == nullptr) {
364 return -1;
365 }
366
367 EnterCriticalSection(&flag_critical_);
368 int local_read_index = read_index_;
369
370 if (buffer_data_[local_read_index].state != 2) {
371 LeaveCriticalSection(&flag_critical_);
372 return -1;
373 }
374
375 *buffer_data = &buffer_data_[local_read_index];
376
377 *time_get = buffer_data_[local_read_index].time;
378 buffer_data_[local_read_index].state = 3;
379 geted_inedx_ = local_read_index;
380
381 if (last_mode_) {
382 if (!allow_overwrite_) {
383 // clear previous
384 int s = local_read_index - 1;
385 int e = local_read_index + 1;
386 if (s == -1) {
387 s = buffer_count_ - 1;
388 }
389 if (e >= buffer_count_) {
390 e = 0;
391 }
392 for (int i = s; i != e;) {
393 if (buffer_data_[i].state != 2) {
394 break;
395 }
396 buffer_data_[i].state = 0;
397
398 i--;
399 if (i == 0) {
400 i = buffer_count_ - 1;
401 }
402 }
403 }
404 }
405 else {
406 read_index_++;
407 if (read_index_ >= buffer_count_) {
408 read_index_ = 0;
409 }
410 }
411
412 LeaveCriticalSection(&flag_critical_);
413
414 return geted_inedx_;
415}
