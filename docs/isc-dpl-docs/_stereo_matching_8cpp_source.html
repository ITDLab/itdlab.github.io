463void StereoMatching::executeMatching(int imghgt, int imgwdt, int depth,
464 unsigned char *pimgref, unsigned char *pimgcmp,
465 float * pblkdsp, int *pblkcrst)
466{
467
468 // 視差ブロックの高さ
469 int stphgt = disparityBlockHeight;
470 // 視差ブロックの幅
471 int stpwdt = disparityBlockWidth;
472 // マッチングブロックの高さ
473 int blkhgt = matchingBlockHeight;
474 // マッチングブロックの幅
475 int blkwdt = matchingBlockWidth;
476
477 // 出力視差ブロック画像の高さ
478 // * 小数切り捨て
479 int imghgtblk = imghgt / stphgt;
480 // 出力視差ブロック画像の幅
481 // * 小数切り捨て
482 int imgwdtblk = imgwdt / stpwdt;
483
484 // コントラスト閾値
485 int crstthr = contrastThreshold;
486 // センサー輝度高解像度モードステータス
487 int crsthrm = contrastHighResoMode;
488
489 // コントラストオフセット
490 int crstofs = 0;
491
492 if (imgwdt == IMG_WIDTH_VM) {
493 crstofs = (int)CONTRAST_OFFSET_VM;
494 }
495 else if (imgwdt == IMG_WIDTH_XC) {
496 crstofs = (int)CONTRAST_OFFSET_XC;
497 }
498
499 // バックマッチング
500 float *pblkbkdsp = NULL;
501 shadeWidth = depth;
502
503 if (enableBackMatching == 1) {
504 memset(bk_block_dsp, 0, imghgt * imgwdt * sizeof(float));
505 pblkbkdsp = bk_block_dsp;
506 shadeWidth = 0;
507 }
508
509 // 視差を取得する
510 getMatchingDisparity(imghgt, imgwdt, depth, crstthr, crstofs, crsthrm,
511 stphgt, stpwdt, blkhgt, blkwdt, imghgtblk, imgwdtblk,
512 pimgref, pimgcmp, pblkdsp, pblkbkdsp, pblkcrst);
513
514 if (enableBackMatching == 1) {
515 blendBothMatchingDisparity(imghgt, imgwdt, imghgtblk, imgwdtblk,
516 backMatchingEvaluationWidth, backMatchingEvaluationRange, backMatchingValidRatio, backMatchingZeroRatio,
517 pblkdsp, pblkbkdsp);
518 }

662void StereoMatching::getDisparityBySSD(int x, int y, int imghgt, int imgwdt, int depth,
663 int crstthr, int crstofs, int crsthrm,
664 int stphgt, int stpwdt, int blkhgt, int blkwdt, int imghgtblk, int imgwdtblk,
665 unsigned char *pimgref, unsigned char *pimgcmp, float * pblkdsp, int *pblkcrst)
666{
667
668 // 視差画像の幅
669 int imgwdtdsp = imgwdt - depth;
670 int jpx = y;
671 int ipx = x;
672 if (ipx >= imgwdtdsp || jpx % stphgt != 0 || ipx % stpwdt != 0) {
673 return;
674 }
675
676 // jblk : マッチングブロックのyインデックス
677 // iblk : マッチングブロックのxインデックス
678 int jblk = jpx / stphgt;
679 int iblk = ipx / stpwdt;
680
681 // ブロック輝度最大値の最小値
682 int bgtmax = BLOCK_BRIGHTNESS_MAX;
683 // 高解像度モードのコントラスト計算のための平均輝度
684 int crstave = CONTRAST_CALC_AVERAGE_FOR_HIGH_RESO;
685 // ブロック平均輝度最小値
686 double minlave = BLOCK_MIN_AVERAGE_BRIGHTNESS;
687 // ブロック内の輝度差の最小値
688 int mindltl = BLOCK_MIN_DELTA_BRIGHTNESS;
689
690 // SSDを保存する配列
691 // 配列サイズはマッチング探索幅
692 float ssd[ISC_IMG_DEPTH_MAX];
693 // SSD最大値を設定
694 float misum = FLT_MAX;
695
696 // 視差値の初期値を設定する
697 int disp = 0;
698
699 // マッチング評価関数
700 // 左右画像の同じ位置iにある輝度の差の二乗を求め，それを領域全体に渡って足し合わせる
701 // SSD = Σ(((右画像の輝度ij - 右画像ブロック輝度平均) - (左画像の輝度ij - 左画像ブロック輝度平均))^2)
702 // 右画像を1画素ずつ探索範囲DEPTHまで右へ動かしてSSDを計算する
703 // 一番小さいSSDのとき，最も類似している
704 // その探索位置を視差値（整数部）とする
705 // 前後のSSDからサブピクセル値（少数部）を算出する
706 int jpxe = jpx + blkhgt;
707 int ipxe = ipx + blkwdt;
708
709 int sumr = 0; // Σ基準画像の輝度ij
710 int sumrr = 0; // Σ(基準画像の輝度ij^2)
711
712 int Lmin = 255; // ブロック内輝度最小値（初期値256階調最大輝度値）
713 int Lmax = 0; // ブロック内輝度最大値（初期値256階調最小輝度値）
714
715 int blkcnt = blkhgt * blkwdt;
716
717 // 右画像ブロック輝度を評価する
718 for (int j = jpx; j < jpxe; j++) {
719 for (int i = ipx; i < ipxe; i++) {
720 int rfx = pimgref[j * imgwdt + i];
721
722 sumr += rfx;
723 sumrr += rfx * rfx;
724
725 if (Lmin > rfx) {
726 Lmin = rfx;
727 }
728 if (Lmax < rfx) {
729 Lmax = rfx;
730 }
731
732 }
733 }
734
735 // コントラストを求める
736 int crst = 0;
737
738 // ブロックの輝度値の平均
739 double Lave = sumr / blkcnt;
740 // ブロック内の輝度差
741 int deltaL = Lmax - Lmin;
742
743 // 以下の場合はコントラスト値はゼロ
744 // ブロックの最小輝度値が閾値未満
745 // ブロック内の輝度差が閾値未満
746 // ブロック平均輝度が閾値未満
747 if (Lmax >= bgtmax && deltaL >= mindltl && Lave >= minlave) {
748 // 輝度線形の場合
749 if (crsthrm == 0) {
750 crst = (int)((deltaL * 1000 - crstofs) / Lave);
751 }
752 // 輝度非線形の場合
753 // High Resolutionモード
754 else {
755 crst = (int)((deltaL * 1000 - crstofs) / crstave);
756 }
757 }
758
759 // コントラストが閾値未満の場合は視差値ゼロにする
760 if (crst < crstthr) {
761 // 視差値
762 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
763 // コントラスト
764 pblkcrst[jblk * imgwdtblk + iblk] = 0;
765 return;
766 }
767
768 for (int k = 0; k < depth; k++) {
769 float sumsq = 0.0;
770
771 int sumc = 0; // Σ比較画像の輝度ij
772 int sumcc = 0; // Σ(比較画像の輝度ij^2)
773 int sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
774
775 for (int j = jpx; j < jpxe; j++) {
776 for (int i = ipx; i < ipxe; i++) {
777 int rfx = pimgref[j * imgwdt + i];
778 int cpx = pimgcmp[j * imgwdt + i + k];
779
780 sumc += cpx;
781 sumcc += cpx * cpx;
782 sumrc += rfx * cpx;
783
784 }
785 }
786 sumsq = sumrr + sumcc - 2 * sumrc - (float)(sumr * sumr + sumc * sumc - 2 * sumr * sumc) / blkcnt;
787
788 // 右画像を1画素ずつ右へ動かしてSSDを計算する
789 // 一番小さいSSDのとき，最も類似している
790 ssd[k] = sumsq;
791 if (sumsq < misum) {
792 misum = sumsq;
793 disp = k;
794 }
795 }
796
797 // 視差値が探索開始スキップ幅：1未満の場合はサブピクセルを算出しない
798 // 視差値が探索幅の上限に達していた場合はサブピクセルを算出しない
799 // 視差値が1未満の場合は視差値ゼロにする
800 // 視差値が探索幅の上限に達していた場合は視差値ゼロにする
801 if (disp < 1 || disp >= (depth - 1)) {
802 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
803 }
804 else {
805 // サブピクセル推定
806 // 折れ線近似
807 // Xsub = (S(-1) - S(1)) / (2 x S(-1) - 2 x S(0)) S(-1) > S(1)
808 // Xsub = (S(1) - S(-1)) / (2 x S(1) - 2 x S(0)) S(-1) ≦ S(1)
809 // 放物線近似
810 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
811 // 折れ線+放物線近似
812 // Xsub = (S(-1) - S(1)) / (S(-1) - S(0) - S(-1) + S(2))
813 // Xsub = (S(1) - S(-1)) / (S(-2) - S(-1) - S(0) + S(1))
814 float sub;
815 float ssdprv = ssd[disp - 1];

824 }

897void StereoMatching::getBothDisparityBySSD(int x, int y, int imghgt, int imgwdt, int depth,
898 int crstthr, int crstofs, /*int bgtmax,*/ int crsthrm,
899 int stphgt, int stpwdt, int blkhgt, int blkwdt, int imghgtblk, int imgwdtblk,
900 unsigned char *pimgref, unsigned char *pimgcmp,
901 float * pblkdsp, float * pblkbkdsp, int *pblkcrst)
902{
903
904 // 視差画像の幅
905 int imgwdtdsp = imgwdt - blkwdt;
906 int jpx = y;
907 int ipx = x;
908 if (ipx >= imgwdtdsp || jpx % stphgt != 0 || ipx % stpwdt != 0) {
909 return;
910 }
911
912 // jblk : マッチングブロックのyインデックス
913 // iblk : マッチングブロックのxインデックス
914 int jblk = jpx / stphgt;
915 int iblk = ipx / stpwdt;
916
917 // ブロック輝度最大値の最小値
918 int bgtmax = BLOCK_BRIGHTNESS_MAX;
919 // 高解像度モードのコントラスト計算のための平均輝度
920 int crstave = CONTRAST_CALC_AVERAGE_FOR_HIGH_RESO;
921 // ブロック平均輝度最小値
922 double minlave = BLOCK_MIN_AVERAGE_BRIGHTNESS;
923 // ブロック内の輝度差の最小値
924 int mindltl = BLOCK_MIN_DELTA_BRIGHTNESS;
925
926 // SSDを保存する配列
927 // 配列サイズはマッチング探索幅
928 float ssd[ISC_IMG_DEPTH_MAX];
929 float bk_ssd[ISC_IMG_DEPTH_MAX];
930 // SSD最大値を設定
931 float misum = FLT_MAX;
932 float bk_misum = FLT_MAX;
933
934 // 視差値の初期値を設定する
935 int disp = 0;
936 int bk_disp = 0;
937
938 // マッチング評価関数
939 // 左右画像の同じ位置iにある輝度の差の二乗を求め，それを領域全体に渡って足し合わせる
940 // SSD = Σ(((右画像の輝度ij - 右画像ブロック輝度平均) - (左画像の輝度ij - 左画像ブロック輝度平均))^2)
941 // 右画像を1画素ずつ探索範囲DEPTHまで右へ動かしてSSDを計算する
942 // 一番小さいSSDのとき，最も類似している
943 // その探索位置を視差値（整数部）とする
944 // 前後のSSDからサブピクセル値（少数部）を算出する
945 int jpxe = jpx + blkhgt;
946 int ipxe = ipx + blkwdt;
947
948 // フォアマッチング
949 int sumr = 0; // Σ基準画像の輝度ij
950 int sumrr = 0; // Σ(基準画像の輝度ij^2)
951 int Lmin = 255; // ブロック内輝度最小値（初期値256階調最大輝度値）
952 int Lmax = 0; // ブロック内輝度最大値（初期値256階調最小輝度値）
953
954 // バックマッチング
955 int bk_sumr = 0; // Σ基準画像の輝度ij
956 int bk_sumrr = 0; // Σ(基準画像の輝度ij^2)
957 int bk_Lmin = 255; // ブロック内輝度最小値（初期値256階調最大輝度値）
958 int bk_Lmax = 0; // ブロック内輝度最大値（初期値256階調最小輝度値）
959
960 int blkcnt = blkhgt * blkwdt;
961
962 // 基準画像ブロック輝度を評価する
963 for (int j = jpx; j < jpxe; j++) {
964 for (int i = ipx; i < ipxe; i++) {
965 int rfx = pimgref[j * imgwdt + i]; // 基準（右）カメラ
966 int cpx = pimgcmp[j * imgwdt + i]; // 比較（左）カメラ
967
968 // フォアマッチング
969 sumr += rfx;
970 sumrr += rfx * rfx;
971 if (Lmin > rfx) {
972 Lmin = rfx;
973 }
974 if (Lmax < rfx) {
975 Lmax = rfx;
976 }
977
978 // バックマッチング
979 bk_sumr += cpx;
980 bk_sumrr += cpx * cpx;
981 if (cpx > bk_Lmax) {
982 bk_Lmax = cpx;
983 }
984 else if (cpx < bk_Lmin) {
985 bk_Lmin = cpx;
986 }
987
988 }
989 }
990
991 // コントラストを求める
992 // フォアマッチング
993 int crst = 0;
994
995 // ブロックの輝度値の平均
996 double Lave = sumr / blkcnt;
997 // ブロック内の輝度差
998 int deltaL = Lmax - Lmin;
999
1000 // 以下の場合はコントラスト値はゼロ
1001 // ブロックの最小輝度値が閾値未満
1002 // ブロック内の輝度差が閾値未満
1003 // ブロック平均輝度が閾値未満
1004 if (Lmax >= bgtmax && deltaL >= mindltl && Lave >= minlave) {
1005 // 輝度線形の場合
1006 if (crsthrm == 0) {
1007 crst = (int)((deltaL * 1000 - crstofs) / Lave);
1008 }
1009 // 輝度非線形の場合
1010 // High Resolutionモード
1011 else {
1012 crst = (int)((deltaL * 1000 - crstofs) / crstave);
1013 }
1014 }
1015
1016 // バックマッチング
1017 int bk_crst = 0;
1018
1019 // ブロックの輝度値の平均
1020 double bk_Lave = bk_sumr / blkcnt;
1021 // ブロック内の輝度差
1022 int bk_deltaL = bk_Lmax - bk_Lmin;
1023
1024 if (bk_Lmax >= bgtmax && bk_deltaL >= mindltl && bk_Lave >= minlave) {
1025 // 輝度線形の場合
1026 if (crsthrm == 0) {
1027 bk_crst = (int)((bk_deltaL * 1000 - crstofs) / bk_Lave);
1028 }
1029 // 輝度非線形の場合
1030 // High Resolutionモード
1031 else {
1032 bk_crst = (int)((bk_deltaL * 1000 - crstofs) / crstave);
1033 }
1034 }
1035
1036 // 探索マージンを求める
1037 // フォアマッチング
1038 int fr_mrgn = imgwdt - (ipx + depth + blkwdt);
1039 // バックマッチング
1040 int bk_mrgn = ipx - depth;
1041 // 探索幅を求める
1042 // フォワード
1043 int fr_depth = depth;
1044 if (fr_mrgn < 0) {
1045 fr_depth = depth + fr_mrgn + 1;
1046 }
1047 // バックワード探索幅
1048 int bk_depth = depth;
1049 if (bk_mrgn < 0) {
1050 bk_depth = depth + bk_mrgn + 1;
1051 }
1052
1053 for (int k = 0; k < depth; k++) {
1054
1055 // フォアマッチング
1056 int sumc = 0; // Σ比較画像の輝度ij
1057 int sumcc = 0; // Σ(比較画像の輝度ij^2)
1058 int sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
1059 // バックマッチング
1060 int bk_sumc = 0; // Σ比較画像の輝度ij
1061 int bk_sumcc = 0; // Σ(比較画像の輝度ij^2)
1062 int bk_sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
1063
1064 for (int j = jpx; j < jpxe; j++) {
1065 for (int i = ipx; i < ipxe; i++) {
1066 // フォアマッチング
1067 if (k < fr_depth) {
1068 int rfx = pimgref[j * imgwdt + i];
1069 int cpx = pimgcmp[j * imgwdt + i + k];
1070 sumc += cpx;
1071 sumcc += cpx * cpx;
1072 sumrc += rfx * cpx;
1073 }
1074 // バックマッチング
1075 if (k < bk_depth) {
1076 int rfx = pimgcmp[j * imgwdt + i];
1077 int cpx = pimgref[j * imgwdt + i - k];
1078 bk_sumc += cpx;
1079 bk_sumcc += cpx * cpx;
1080 bk_sumrc += rfx * cpx;
1081 }
1082 }
1083 }
1084 // フォアマッチング
1085 float sumsq = sumrr + sumcc - 2 * sumrc - (float)(sumr * sumr + sumc * sumc - 2 * sumr * sumc) / blkcnt;
1086 // バックマッチング
1087 float bk_sumsq = bk_sumrr + bk_sumcc - 2 * bk_sumrc
1088 - (float)(bk_sumr * bk_sumr + bk_sumc * bk_sumc - 2 * bk_sumr * bk_sumc) / blkcnt;
1089
1090 // 右画像を1画素ずつ右へ動かしてSSDを計算する
1091 // 一番小さいSSDのとき，最も類似している
1092 // フォアマッチング
1093 ssd[k] = sumsq;
1094 if (sumsq < misum) {
1095 misum = sumsq;
1096 disp = k;
1097 }
1098 // バックマッチング
1099 bk_ssd[k] = bk_sumsq;
1100 if (bk_sumsq < bk_misum) {
1101 bk_misum = bk_sumsq;
1102 bk_disp = k;
1103 }
1104 }
1105
1106 // 視差値が探索開始スキップ幅：1未満の場合はサブピクセルを算出しない
1107 // 視差値が探索幅の上限に達していた場合はサブピクセルを算出しない
1108 // 視差値が1未満の場合は視差値ゼロにする
1109 // 視差値が探索幅の上限に達していた場合は視差値ゼロにする
1110 // コントラストが閾値未満の場合は視差値ゼロにする
1111 float sub;
1112 float ssdprv;
1113 float ssdcnt;
1114 float ssdnxt;
1115 // フォアマッチング
1116 if (fr_depth < 3 || disp < 1 || disp >= (fr_depth - 1) || crst < crstthr) {
1117 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
1118 crst = 0;
1119 }
1120 else {
1121 // 放物線近似
1122 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
1123 ssdprv = ssd[disp - 1];
1124 ssdcnt = ssd[disp];
1125 ssdnxt = ssd[disp + 1];
1126 // 放物線近似
1127 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
1128 // 視差値を保存する
1129 pblkdsp[jblk * imgwdtblk + iblk] = disp + sub;
1130 }
1131 // バックマッチング
1132 if (bk_depth >= 3 && bk_disp >= 1 && bk_disp < (bk_depth - 1) && bk_crst >= crstthr) {
1133 // 放物線近似
1134 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
1135 ssdprv = bk_ssd[bk_disp - 1];
1136 ssdcnt = bk_ssd[bk_disp];
1137 ssdnxt = bk_ssd[bk_disp + 1];
1138 // 放物線近似
1139 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
1140 float bk_disp_sub = bk_disp + sub;
1141 // バックマッチングの結果を基準画像の座標へ展開
1142 // 視差ブロック番号
1143 int bk_iblk = (int)((ipx - bk_disp_sub) / stpwdt);
1144 // 視差値を保存する
1145 pblkbkdsp[jblk * imgwdtblk + bk_iblk] = bk_disp_sub;
1146 }
1147
1148 // コントラストを保存する
1149 pblkcrst[jblk * imgwdtblk + iblk] = crst;
1150
1151}

1167void StereoMatching::blendBothMatchingDisparity(int imghgt, int imgwdt, int imghgtblk, int imgwdtblk,

1173 // バックマッチング評価視差正当ブロック数
1174 int bkvldnum = (int)((bkevlblk * bkvldrt) / 100);
1175 // バックマッチング評価視差ゼロブロック数

1177

1179 for (int id = 0; id < imgwdtblk; id++) {
1180
1181 if (jd < bkevlwdt || jd >= imghgtblk - bkevlwdt || id < bkevlwdt || id >= imgwdtblk - bkevlwdt) {
1182 pblkdsp[jd * imgwdtblk + id] = 0.0;
1183 continue;
1184 }
1185

1187 if (disp != 0.0) {

1189 int bk_evlcnt = 0;
1190 for (int j = jd - bkevlwdt; j <= jd + bkevlwdt; j++) {
1191 for (int i = id - bkevlwdt; i <= id + bkevlwdt; i++) {

1194 // 視差値ゼロをカウントする
1195 if (bk_disp == 0.0) {
1196 bk_zrcnt++;
1197 }
1198 // 指定差以下の視差をカウントする
1199 if (dispdiff <= (float)bkevlrng) {
1200 bk_evlcnt++;
1201 }
1202 }
1203 }
1204 // 視差値ゼロの数が指定以上の場合は視差値ゼロにする
1205 if (bk_zrcnt >= bkzrnum) {
1206 pblkdsp[jd * imgwdtblk + id] = 0.0;
1207 }
1208 // 範囲内視差差の数が指定以上の場合は視差を残す
1209 // 指定未満の場合は視差値ゼロにする
1210 else if (bk_evlcnt < bkvldnum) {
1211 pblkdsp[jd * imgwdtblk + id] = 0.0;
1212 }
1213 }
1214 }
1215 }
