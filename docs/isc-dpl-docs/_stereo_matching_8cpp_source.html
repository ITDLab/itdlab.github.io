474void StereoMatching::executeMatching(int imghgt, int imgwdt, int depth,
475 unsigned char *pimgref, unsigned char *pimgcmp,
476 float * pblkdsp, int *pblkcrst)
477{
478
479 // 視差ブロックの高さ
480 int stphgt = disparityBlockHeight;
481 // 視差ブロックの幅
482 int stpwdt = disparityBlockWidth;
483 // マッチングブロックの高さ
484 int blkhgt = matchingBlockHeight;
485 // マッチングブロックの幅
486 int blkwdt = matchingBlockWidth;
487
488 // 出力視差ブロック画像の高さ
489 // * 小数切り捨て
490 int imghgtblk = imghgt / stphgt;
491 // 出力視差ブロック画像の幅
492 // * 小数切り捨て
493 int imgwdtblk = imgwdt / stpwdt;
494
495 // コントラスト閾値
496 int crstthr = contrastThreshold;
497 // 階調補正モードステータス
498 int grdcrct = gradationCorrectionMode;
499
500 // コントラストオフセット
501 int crstofs = 0;
502
503 if (imgwdt == IMG_WIDTH_VM) {
504 crstofs = (int)CONTRAST_OFFSET_VM;
505 }
506 else if (imgwdt == IMG_WIDTH_XC) {
507 crstofs = (int)CONTRAST_OFFSET_XC;
508 // 階調補正モードの場合
509 if (grdcrct == 1) {
510 crstofs = (int)(CONTRAST_OFFSET_XC * CONTRAST_XC_DRADATION_FACTOR);
511 }
512 }
513
514 // バックマッチング
515 float *pblkbkdsp = NULL;
516 shadeWidth = depth;
517
518 if (enableBackMatching == 1) {
519 memset(bk_block_dsp, 0, imghgt * imgwdt * sizeof(float));
520 pblkbkdsp = bk_block_dsp;
521 shadeWidth = 0;
522 }
523
524 // 視差を取得する
525 getMatchingDisparity(imghgt, imgwdt, depth, crstthr, crstofs, grdcrct,
526 stphgt, stpwdt, blkhgt, blkwdt, imghgtblk, imgwdtblk,
527 pimgref, pimgcmp, pblkdsp, pblkbkdsp, pblkcrst);
528
529 if (enableBackMatching == 1) {
530 blendBothMatchingDisparity(imghgt, imgwdt, imghgtblk, imgwdtblk,
531 backMatchingEvaluationWidth, backMatchingEvaluationRange, backMatchingValidRatio, backMatchingZeroRatio,
532 pblkdsp, pblkbkdsp);
533 }

680void StereoMatching::getDisparityBySSD(int x, int y, int imghgt, int imgwdt, int depth,
681 int crstthr, int crstofs, int grdcrct,
682 int stphgt, int stpwdt, int blkhgt, int blkwdt, int imghgtblk, int imgwdtblk,
683 unsigned char *pimgref, unsigned char *pimgcmp, float * pblkdsp, int *pblkcrst)
684{
685
686 // 視差画像の幅
687 int imgwdtdsp = imgwdt - depth;
688 int jpx = y;
689 int ipx = x;
690 if (ipx >= imgwdtdsp || jpx % stphgt != 0 || ipx % stpwdt != 0) {
691 return;
692 }
693
694 // jblk : マッチングブロックのyインデックス
695 // iblk : マッチングブロックのxインデックス
696 int jblk = jpx / stphgt;
697 int iblk = ipx / stpwdt;
698
699 // ブロック輝度最大値の最小値
700 int bgtmax = BLOCK_BRIGHTNESS_MAX;
701 // 高解像度モードのコントラスト計算のための平均輝度
702 int crstave = CONTRAST_CALC_AVERAGE_FOR_HIGH_RESO;
703 // ブロック平均輝度最小値
704 double minlave = BLOCK_MIN_AVERAGE_BRIGHTNESS;
705 // ブロック内の輝度差の最小値
706 int mindltl = BLOCK_MIN_DELTA_BRIGHTNESS;
707
708 // SSDを保存する配列
709 // 配列サイズはマッチング探索幅
710 float ssd[ISC_IMG_DEPTH_MAX];
711 // SSD最大値を設定
712 float misum = FLT_MAX;
713
714 // 視差値の初期値を設定する
715 int disp = 0;
716
717 // マッチング評価関数
718 // 左右画像の同じ位置iにある輝度の差の二乗を求め，それを領域全体に渡って足し合わせる
719 // SSD = Σ(((右画像の輝度ij - 右画像ブロック輝度平均) - (左画像の輝度ij - 左画像ブロック輝度平均))^2)
720 // 右画像を1画素ずつ探索範囲DEPTHまで右へ動かしてSSDを計算する
721 // 一番小さいSSDのとき，最も類似している
722 // その探索位置を視差値（整数部）とする
723 // 前後のSSDからサブピクセル値（少数部）を算出する
724 int jpxe = jpx + blkhgt;
725 int ipxe = ipx + blkwdt;
726
727 int sumr = 0; // Σ基準画像の輝度ij
728 int sumrr = 0; // Σ(基準画像の輝度ij^2)
729 double Lsumr = 0.0; // Σ基準画像の輝度ij
730 double Lmin = 255.0; // ブロック内輝度最小値（初期値256階調最大輝度値）
731 double Lmax = 0.0; // ブロック内輝度最大値（初期値256階調最小輝度値）
732 int blkcnt = blkhgt * blkwdt;
733
734 // 右画像ブロック輝度を評価する
735 for (int j = jpx; j < jpxe; j++) {
736 for (int i = ipx; i < ipxe; i++) {
737 double rfx = (double)pimgref[j * imgwdt + i];
738 double xrfx = rfx * rfx;
739
740 sumr += (int)rfx;
741 sumrr += (int)xrfx;
742
743 // 階調補正モードオンの場合
744 // 補正前の値へ変換する
745 if (grdcrct == 1) {
746 rfx = xrfx / 255;
747 }
748 Lsumr += rfx;
749 if (Lmin > rfx) {
750 Lmin = rfx;
751 }
752 if (Lmax < rfx) {
753 Lmax = rfx;
754 }
755
756 }
757 }
758
759 // コントラストを求める
760 int crst = 0;
761
762 // ブロックの輝度値の平均
763 double Lave = Lsumr / blkcnt;
764 // ブロック内の輝度差
765 double deltaL = Lmax - Lmin;
766
767 // 以下の場合はコントラスト値はゼロ
768 // ブロックの最小輝度値が閾値未満
769 // ブロック内の輝度差が閾値未満
770 // ブロック平均輝度が閾値未満
771 if (Lmax >= (double)bgtmax && deltaL >= (double)mindltl && Lave >= minlave) {
772 crst = (int)((deltaL * 1000 - crstofs) / Lave);
773 }
774
775 // コントラストが閾値未満の場合は視差値ゼロにする
776 if (crst < crstthr) {
777 // 視差値
778 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
779 // コントラスト
780 pblkcrst[jblk * imgwdtblk + iblk] = 0;
781 return;
782 }
783
784 for (int k = 0; k < depth; k++) {
785 float sumsq = 0.0;
786
787 int sumc = 0; // Σ比較画像の輝度ij
788 int sumcc = 0; // Σ(比較画像の輝度ij^2)
789 int sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
790
791 for (int j = jpx; j < jpxe; j++) {
792 for (int i = ipx; i < ipxe; i++) {
793 int rfx = pimgref[j * imgwdt + i];
794 int cpx = pimgcmp[j * imgwdt + i + k];
795
796 sumc += cpx;
797 sumcc += cpx * cpx;
798 sumrc += rfx * cpx;
799
800 }
801 }
802 sumsq = sumrr + sumcc - 2 * sumrc - (float)(sumr * sumr + sumc * sumc - 2 * sumr * sumc) / blkcnt;
803
804 // 右画像を1画素ずつ右へ動かしてSSDを計算する
805 // 一番小さいSSDのとき，最も類似している
806 ssd[k] = sumsq;
807 if (sumsq < misum) {
808 misum = sumsq;
809 disp = k;
810 }
811 }
812
813 // 視差値が探索開始スキップ幅：1未満の場合はサブピクセルを算出しない
814 // 視差値が探索幅の上限に達していた場合はサブピクセルを算出しない
815 // 視差値が1未満の場合は視差値ゼロにする
816 // 視差値が探索幅の上限に達していた場合は視差値ゼロにする
817 if (disp < 1 || disp >= (depth - 1)) {
818 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
819 }
820 else {
821 // サブピクセル推定
822 // 折れ線近似
823 // Xsub = (S(-1) - S(1)) / (2 x S(-1) - 2 x S(0)) S(-1) > S(1)
824 // Xsub = (S(1) - S(-1)) / (2 x S(1) - 2 x S(0)) S(-1) ≦ S(1)
825 // 放物線近似
826 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
827 // 折れ線+放物線近似
828 // Xsub = (S(-1) - S(1)) / (S(-1) - S(0) - S(-1) + S(2))
829 // Xsub = (S(1) - S(-1)) / (S(-2) - S(-1) - S(0) + S(1))
830 float sub;
831 float ssdprv = ssd[disp - 1];
832 float ssdcnt = ssd[disp];
833 float ssdnxt = ssd[disp + 1];
834
835 // 放物線近似
836 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
837
838 pblkdsp[jblk * imgwdtblk + iblk] = disp + sub;
839
840 }
841
842 // コントラストを保存する
843 pblkcrst[jblk * imgwdtblk + iblk] = crst;
844
845}

913void StereoMatching::getBothDisparityBySSD(int x, int y, int imghgt, int imgwdt, int depth,
914 int crstthr, int crstofs, int grdcrct,
915 int stphgt, int stpwdt, int blkhgt, int blkwdt, int imghgtblk, int imgwdtblk,
916 unsigned char *pimgref, unsigned char *pimgcmp,
917 float * pblkdsp, float * pblkbkdsp, int *pblkcrst)
918{
919
920 // 視差画像の幅
921 int imgwdtdsp = imgwdt - blkwdt;
922 int jpx = y;
923 int ipx = x;
924 if (ipx >= imgwdtdsp || jpx % stphgt != 0 || ipx % stpwdt != 0) {
925 return;
926 }
927
928 // jblk : マッチングブロックのyインデックス
929 // iblk : マッチングブロックのxインデックス
930 int jblk = jpx / stphgt;
931 int iblk = ipx / stpwdt;
932
933 // ブロック輝度最大値の最小値
934 int bgtmax = BLOCK_BRIGHTNESS_MAX;
935 // 高解像度モードのコントラスト計算のための平均輝度
936 int crstave = CONTRAST_CALC_AVERAGE_FOR_HIGH_RESO;
937 // ブロック平均輝度最小値
938 double minlave = BLOCK_MIN_AVERAGE_BRIGHTNESS;
939 // ブロック内の輝度差の最小値
940 int mindltl = BLOCK_MIN_DELTA_BRIGHTNESS;
941
942 // SSDを保存する配列
943 // 配列サイズはマッチング探索幅
944 float ssd[ISC_IMG_DEPTH_MAX];
945 float bk_ssd[ISC_IMG_DEPTH_MAX];
946 // SSD最大値を設定
947 float misum = FLT_MAX;
948 float bk_misum = FLT_MAX;
949
950 // 視差値の初期値を設定する
951 int disp = 0;
952 int bk_disp = 0;
953
954 // マッチング評価関数
955 // 左右画像の同じ位置iにある輝度の差の二乗を求め，それを領域全体に渡って足し合わせる
956 // SSD = Σ(((右画像の輝度ij - 右画像ブロック輝度平均) - (左画像の輝度ij - 左画像ブロック輝度平均))^2)
957 // 右画像を1画素ずつ探索範囲DEPTHまで右へ動かしてSSDを計算する
958 // 一番小さいSSDのとき，最も類似している
959 // その探索位置を視差値（整数部）とする
960 // 前後のSSDからサブピクセル値（少数部）を算出する
961 int jpxe = jpx + blkhgt;
962 int ipxe = ipx + blkwdt;
963
964 // フォアマッチング
965 int sumr = 0; // Σ基準画像の輝度ij
966 int sumrr = 0; // Σ(基準画像の輝度ij^2)
967 double Lsumr = 0.0; // Σ基準画像の輝度ij
968 double Lmin = 255.0; // ブロック内輝度最小値（初期値256階調最大輝度値）
969 double Lmax = 0.0; // ブロック内輝度最大値（初期値256階調最小輝度値）
970
971 // バックマッチング
972 int bk_sumr = 0; // Σ基準画像の輝度ij
973 int bk_sumrr = 0; // Σ(基準画像の輝度ij^2)
974 double bk_Lsumr = 0.0; // Σ基準画像の輝度ij
975 double bk_Lmin = 255.0; // ブロック内輝度最小値（初期値256階調最大輝度値）
976 double bk_Lmax = 0.0; // ブロック内輝度最大値（初期値256階調最小輝度値）
977
978 int blkcnt = blkhgt * blkwdt;
979
980 // 基準画像ブロック輝度を評価する
981 for (int j = jpx; j < jpxe; j++) {
982 for (int i = ipx; i < ipxe; i++) {
983
984 double rfx = (double)pimgref[j * imgwdt + i]; // 基準（右）カメラ
985 double xrfx = rfx * rfx;
986 double cpx = (double)pimgcmp[j * imgwdt + i]; // 比較（左）カメラ
987 double xcpx = cpx * cpx;
988
989 // フォアマッチング
990 sumr += (int)rfx;
991 sumrr += (int)xrfx;
992 // バックマッチング
993 bk_sumr += (int)cpx;
994 bk_sumrr += (int)xcpx;
995
996 // 階調補正モードオンの場合
997 // 補正前の値へ変換する
998 if (grdcrct == 1) {
999 rfx = xrfx / 255;
1000 cpx = xcpx / 255;
1001 }
1002 // フォアマッチング
1003 Lsumr += rfx;
1004 if (Lmin > rfx) {
1005 Lmin = rfx;
1006 }
1007 if (Lmax < rfx) {
1008 Lmax = rfx;
1009 }
1010 // バックマッチング
1011 bk_Lsumr += cpx;
1012 if (cpx > bk_Lmax) {
1013 bk_Lmax = cpx;
1014 }
1015 else if (cpx < bk_Lmin) {
1016 bk_Lmin = cpx;
1017 }
1018
1019 }
1020 }
1021
1022 // コントラストを求める
1023 // フォアマッチング
1024 int crst = 0;
1025
1026 // ブロックの輝度値の平均
1027 double Lave = Lsumr / blkcnt;
1028 // ブロック内の輝度差
1029 double deltaL = Lmax - Lmin;
1030
1031 // 以下の場合はコントラスト値はゼロ
1032 // ブロックの最小輝度値が閾値未満
1033 // ブロック内の輝度差が閾値未満
1034 // ブロック平均輝度が閾値未満
1035 if (Lmax >= (double)bgtmax && deltaL >= (double)mindltl && Lave >= minlave) {
1036 crst = (int)((deltaL * 1000 - crstofs) / Lave);
1037 }
1038
1039 // バックマッチング
1040 int bk_crst = 0;
1041
1042 // ブロックの輝度値の平均
1043 double bk_Lave = bk_Lsumr / blkcnt;
1044 // ブロック内の輝度差
1045 double bk_deltaL = bk_Lmax - bk_Lmin;
1046
1047 if (bk_Lmax >= (double)bgtmax && bk_deltaL >= (double)mindltl && bk_Lave >= minlave) {
1048 bk_crst = (int)((bk_deltaL * 1000 - crstofs) / bk_Lave);
1049 }
1050
1051 // 探索マージンを求める
1052 // フォアマッチング
1053 int fr_mrgn = imgwdt - (ipx + depth + blkwdt);
1054 // バックマッチング
1055 int bk_mrgn = ipx - depth;
1056 // 探索幅を求める
1057 // フォワード
1058 int fr_depth = depth;
1059 if (fr_mrgn < 0) {
1060 fr_depth = depth + fr_mrgn + 1;
1061 }
1062 // バックワード探索幅
1063 int bk_depth = depth;
1064 if (bk_mrgn < 0) {
1065 bk_depth = depth + bk_mrgn + 1;
1066 }
1067
1068 for (int k = 0; k < depth; k++) {
1069
1070 // フォアマッチング
1071 int sumc = 0; // Σ比較画像の輝度ij
1072 int sumcc = 0; // Σ(比較画像の輝度ij^2)
1073 int sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
1074 // バックマッチング
1075 int bk_sumc = 0; // Σ比較画像の輝度ij
1076 int bk_sumcc = 0; // Σ(比較画像の輝度ij^2)
1077 int bk_sumrc = 0; // Σ(基準画像の輝度ij*比較画像の輝度ij)
1078
1079 for (int j = jpx; j < jpxe; j++) {
1080 for (int i = ipx; i < ipxe; i++) {
1081 // フォアマッチング
1082 if (k < fr_depth) {
1083 int rfx = pimgref[j * imgwdt + i];
1084 int cpx = pimgcmp[j * imgwdt + i + k];
1085 sumc += cpx;
1086 sumcc += cpx * cpx;
1087 sumrc += rfx * cpx;
1088 }
1089 // バックマッチング
1090 if (k < bk_depth) {
1091 int rfx = pimgcmp[j * imgwdt + i];
1092 int cpx = pimgref[j * imgwdt + i - k];
1093 bk_sumc += cpx;
1094 bk_sumcc += cpx * cpx;
1095 bk_sumrc += rfx * cpx;
1096 }
1097 }
1098 }
1099 // フォアマッチング
1100 float sumsq = sumrr + sumcc - 2 * sumrc - (float)(sumr * sumr + sumc * sumc - 2 * sumr * sumc) / blkcnt;
1101 // バックマッチング
1102 float bk_sumsq = bk_sumrr + bk_sumcc - 2 * bk_sumrc
1103 - (float)(bk_sumr * bk_sumr + bk_sumc * bk_sumc - 2 * bk_sumr * bk_sumc) / blkcnt;
1104
1105 // 右画像を1画素ずつ右へ動かしてSSDを計算する
1106 // 一番小さいSSDのとき，最も類似している
1107 // フォアマッチング
1108 ssd[k] = sumsq;
1109 if (sumsq < misum) {
1110 misum = sumsq;
1111 disp = k;
1112 }
1113 // バックマッチング
1114 bk_ssd[k] = bk_sumsq;
1115 if (bk_sumsq < bk_misum) {
1116 bk_misum = bk_sumsq;
1117 bk_disp = k;
1118 }
1119 }
1120
1121 // 視差値が探索開始スキップ幅：1未満の場合はサブピクセルを算出しない
1122 // 視差値が探索幅の上限に達していた場合はサブピクセルを算出しない
1123 // 視差値が1未満の場合は視差値ゼロにする
1124 // 視差値が探索幅の上限に達していた場合は視差値ゼロにする
1125 // コントラストが閾値未満の場合は視差値ゼロにする
1126 float sub;
1127 float ssdprv;
1128 float ssdcnt;
1129 float ssdnxt;
1130 // フォアマッチング
1131 if (fr_depth < 3 || disp < 1 || disp >= (fr_depth - 1) || crst < crstthr) {
1132 pblkdsp[jblk * imgwdtblk + iblk] = 0.0;
1133 crst = 0;
1134 }
1135 else {
1136 // 放物線近似
1137 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
1138 ssdprv = ssd[disp - 1];
1139 ssdcnt = ssd[disp];
1140 ssdnxt = ssd[disp + 1];
1141 // 放物線近似
1142 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
1143 // 視差値を保存する
1144 pblkdsp[jblk * imgwdtblk + iblk] = disp + sub;
1145 }
1146 // バックマッチング
1147 if (bk_depth >= 3 && bk_disp >= 1 && bk_disp < (bk_depth - 1) && bk_crst >= crstthr) {
1148 // 放物線近似
1149 // Xsub = (S(1) - S(-1)) / (2 x S(-1) - 4 x S(0) + 2 x S(1))
1150 ssdprv = bk_ssd[bk_disp - 1];
1151 ssdcnt = bk_ssd[bk_disp];
1152 ssdnxt = bk_ssd[bk_disp + 1];
1153 // 放物線近似
1154 sub = (ssdprv - ssdnxt) / (2 * ssdprv - 4 * ssdcnt + 2 * ssdnxt);
1155 float bk_disp_sub = bk_disp + sub;
1156 // バックマッチングの結果を基準画像の座標へ展開
1157 // 視差ブロック番号
1158 int bk_iblk = (int)((ipx - bk_disp_sub) / stpwdt);
1159 // 視差値を保存する
1160 pblkbkdsp[jblk * imgwdtblk + bk_iblk] = bk_disp_sub;
1161 }
1162
1163 // コントラストを保存する
1164 pblkcrst[jblk * imgwdtblk + iblk] = crst;
1165
1166}

1224 // 指定未満の場合は視差値ゼロにする
1225 else if (bk_evlcnt < bkvldnum) {

1228 }
1229 }
