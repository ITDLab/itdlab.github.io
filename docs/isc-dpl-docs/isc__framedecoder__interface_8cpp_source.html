117int IscFramedecoderInterface::Initialize(IscDataProcModuleConfiguration* isc_data_proc_module_configuration)
118{
119
120 swprintf_s(isc_data_proc_module_configuration_.configuration_file_path, L"%s", isc_data_proc_module_configuration->configuration_file_path);
121 swprintf_s(isc_data_proc_module_configuration_.log_file_path, L"%s", isc_data_proc_module_configuration->log_file_path);
122
123 isc_data_proc_module_configuration_.log_level = isc_data_proc_module_configuration->log_level;
124 isc_data_proc_module_configuration_.isc_camera_model = isc_data_proc_module_configuration->isc_camera_model;
125 isc_data_proc_module_configuration_.max_image_width = isc_data_proc_module_configuration->max_image_width;
126 isc_data_proc_module_configuration_.max_image_height = isc_data_proc_module_configuration->max_image_height;
127
128 switch (isc_data_proc_module_configuration_.isc_camera_model) {
129 case IscCameraModel::kVM:
130 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_VM.ini", isc_data_proc_module_configuration_.configuration_file_path);
131 break;
132
133 case IscCameraModel::kXC:
134 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_XC.ini", isc_data_proc_module_configuration_.configuration_file_path);
135 break;
136
137 case IscCameraModel::k4K:
138 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_4K.ini", isc_data_proc_module_configuration_.configuration_file_path);
139 break;
140
141 case IscCameraModel::k4KA:
142 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_4KA.ini", isc_data_proc_module_configuration_.configuration_file_path);
143 break;
144
145 case IscCameraModel::k4KJ:
146 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_4KJ.ini", isc_data_proc_module_configuration_.configuration_file_path);
147 break;
148
149 default:
150 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter.ini", isc_data_proc_module_configuration_.configuration_file_path);
151 break;
152 }
153
154 int ret = LoadParameterFromFile(parameter_file_name_, &frame_decoder_parameters_);
155 if (ret != DPC_E_OK) {
156 if (ret == DPCPROCESS_E_FILE_NOT_FOUND) {
157 // Since the file does not exist, it will be created by default and continue
158 SaveParameterToFile(parameter_file_name_, &frame_decoder_parameters_);
159 }
160 else {
161 return ret;
162 }
163 }
164
165 ret = SetParameterToFrameDecoderModule(&frame_decoder_parameters_);
166 if (ret != DPC_E_OK) {
167 return ret;
168 }
169
170 // get work
171 size_t image_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
172 size_t depth_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
173
174 for (int i = 0; i < 4; i++) {
175 work_buffers_.buff_image[i].width = 0;
176 work_buffers_.buff_image[i].height = 0;
177 work_buffers_.buff_image[i].channel_count = 0;
178 work_buffers_.buff_image[i].image = new unsigned char[image_size];
179 memset(work_buffers_.buff_image[i].image, 0, image_size);
180
181 work_buffers_.buff_depth[i].width = 0;
182 work_buffers_.buff_depth[i].height = 0;
183 work_buffers_.buff_depth[i].image = new float[depth_size];
184 memset(work_buffers_.buff_depth[i].image, 0, depth_size * sizeof(float));
185 }
186
187 // initialize Framedecoder
188 ISCFrameDecoder::initialize(isc_data_proc_module_configuration_.max_image_height, isc_data_proc_module_configuration_.max_image_width);
189 ISCFrameDecoder::createAveragingThread();
190
191 // Check if OpenCL is available. Enable it if it is available.
192 if (frame_decoder_parameters_.system_parameter.enabled_opencl_for_avedisp && cv::ocl::haveOpenCL()) {
193 // it can use openCL
194 //cv::String build_info_str = cv::getBuildInformation();
195 //OutputDebugStringA(build_info_str.c_str());
196 ISCFrameDecoder::setUseOpenCLForAveragingDisparity(1);
197 }
198 else {
199 ISCFrameDecoder::setUseOpenCLForAveragingDisparity(0);
200 }
201
202 return DPC_E_OK;
203}

428 frame_decoder_parameters->averaging_parameter.vldrt,
429 frame_decoder_parameters->averaging_parameter.reprt);
430
431 ISCFrameDecoder::setAveragingBlockWeight(
432 frame_decoder_parameters->averaging_block_weight_parameter.cntwgt,
433 frame_decoder_parameters->averaging_block_weight_parameter.nrwgt,
434 frame_decoder_parameters->averaging_block_weight_parameter.rndwgt);
435
436 ISCFrameDecoder::setComplementParameter(
437 frame_decoder_parameters->complement_parameter.enb,
438 frame_decoder_parameters->complement_parameter.lowlmt,

444 frame_decoder_parameters->complement_parameter.hlfil,
445 frame_decoder_parameters->complement_parameter.hlsz);

463 work_buffers_.buff_image[i].width = 0;

465 work_buffers_.buff_image[i].channel_count = 0;

470 work_buffers_.buff_depth[i].height = 0;
471 delete[] work_buffers_.buff_depth[i].image;
472 work_buffers_.buff_depth[i].image = nullptr;
473 }
474
475 return DPC_E_OK;
476}
