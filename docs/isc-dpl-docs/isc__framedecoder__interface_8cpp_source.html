92int IscFramedecoderInterface::Initialize(IscDataProcModuleConfiguration* isc_data_proc_module_configuration)
93{
94
95 swprintf_s(isc_data_proc_module_configuration_.configuration_file_path, L"%s", isc_data_proc_module_configuration->configuration_file_path);
96 swprintf_s(isc_data_proc_module_configuration_.log_file_path, L"%s", isc_data_proc_module_configuration->log_file_path);
97
98 isc_data_proc_module_configuration_.log_level = isc_data_proc_module_configuration->log_level;
99 isc_data_proc_module_configuration_.isc_camera_model = isc_data_proc_module_configuration->isc_camera_model;
100 isc_data_proc_module_configuration_.max_image_width = isc_data_proc_module_configuration->max_image_width;
101 isc_data_proc_module_configuration_.max_image_height = isc_data_proc_module_configuration->max_image_height;
102
103 switch (isc_data_proc_module_configuration_.isc_camera_model) {
104 case IscCameraModel::kVM:
105 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_VM.ini", isc_data_proc_module_configuration_.configuration_file_path);
106 break;
107
108 case IscCameraModel::kXC:
109 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_XC.ini", isc_data_proc_module_configuration_.configuration_file_path);
110 break;
111
112 case IscCameraModel::k4K:
113 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_4K.ini", isc_data_proc_module_configuration_.configuration_file_path);
114 break;
115
116 case IscCameraModel::k4KA:
117 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_4KA.ini", isc_data_proc_module_configuration_.configuration_file_path);
118 break;
119
120 case IscCameraModel::k4KJ:
121 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter_4KJ.ini", isc_data_proc_module_configuration_.configuration_file_path);
122 break;
123
124 default:
125 swprintf_s(parameter_file_name_, L"%s\\FrameDecoderParameter.ini", isc_data_proc_module_configuration_.configuration_file_path);
126 break;
127 }
128
129 int ret = LoadParameterFromFile(parameter_file_name_, &frame_decoder_parameters_);
130 if (ret != DPC_E_OK) {
131 if (ret == DPCPROCESS_E_FILE_NOT_FOUND) {
132 // Since the file does not exist, it will be created by default and continue
133 SaveParameterToFile(parameter_file_name_, &frame_decoder_parameters_);
134 }
135 else {
136 return ret;
137 }
138 }
139
140 ret = SetParameterToFrameDecoderModule(&frame_decoder_parameters_);
141 if (ret != DPC_E_OK) {
142 return ret;
143 }
144
145 // get work
146 size_t image_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
147 size_t depth_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
148
149 for (int i = 0; i < 8; i++) {
150 work_buffers_.buff_image[i].width = 0;
151 work_buffers_.buff_image[i].height = 0;
152 work_buffers_.buff_image[i].channel_count = 0;
153 work_buffers_.buff_image[i].image = new unsigned char[image_size];
154 memset(work_buffers_.buff_image[i].image, 0, image_size);
155
156 work_buffers_.buff_depth[i].width = 0;
157 work_buffers_.buff_depth[i].height = 0;
158 work_buffers_.buff_depth[i].image = new float[depth_size];
159 memset(work_buffers_.buff_depth[i].image, 0, depth_size * sizeof(float));
160
161 work_buffers_.buff_int_image[i].width = 0;
162 work_buffers_.buff_int_image[i].height = 0;
163 work_buffers_.buff_int_image[i].image = new int[image_size];
164 memset(work_buffers_.buff_int_image[i].image, 0, image_size * sizeof(int));
165 }
166
167 // initialize Framedecoder
168 ISCFrameDecoder::initialize(isc_data_proc_module_configuration_.max_image_height, isc_data_proc_module_configuration_.max_image_width);
169
170 return DPC_E_OK;
171}

181int IscFramedecoderInterface::LoadParameterFromFile(const wchar_t* file_name, FrameDecoderParameters* frame_decoder_parameters)
182{
183
184 bool is_exists = PathFileExists(file_name) == TRUE ? true : false;
185 if (!is_exists) {
186 return DPCPROCESS_E_FILE_NOT_FOUND;
187 }
188
189 wchar_t returned_string[1024] = {};
190
191 // Decode
192 GetPrivateProfileString(L"DECODE", L"crstthr", L"0", returned_string, sizeof(returned_string) / sizeof(wchar_t), file_name);
193 frame_decoder_parameters->decode_parameter.crstthr = _wtoi(returned_string);
194
195 GetPrivateProfileString(L"DECODE", L"crsthrm", L"0", returned_string, sizeof(returned_string) / sizeof(wchar_t), file_name);
196 frame_decoder_parameters->decode_parameter.crsthrm = _wtoi(returned_string);
197
198 // DisparityLimitationParameter
199 GetPrivateProfileString(L"DISPARITY_LIMITATION", L"limit", L"0", returned_string, sizeof(returned_string) / sizeof(wchar_t), file_name);
200 frame_decoder_parameters->disparity_limitation_parameter.limit = _wtoi(returned_string);
201
202 GetPrivateProfileString(L"DISPARITY_LIMITATION", L"lower", L"0", returned_string, sizeof(returned_string) / sizeof(wchar_t), file_name);
203 frame_decoder_parameters->disparity_limitation_parameter.lower = _wtoi(returned_string);
204
205 GetPrivateProfileString(L"DISPARITY_LIMITATION", L"upper", L"0", returned_string, sizeof(returned_string) / sizeof(wchar_t), file_name);
206 frame_decoder_parameters->disparity_limitation_parameter.upper = _wtoi(returned_string);
207
208
209 return DPC_E_OK;
210}

488{
489

491
492 // DecodeParameter

494 ParseParameterSet(&isc_data_proc_module_parameter->parameter_set[index++], &frame_decoder_parameters_.decode_parameter.crsthrm);

499 ParseParameterSet(&isc_data_proc_module_parameter->parameter_set[index++], &frame_decoder_parameters_.disparity_limitation_parameter.upper);
500
501 parameter_update_request_ = true;
502
503 // file
504 if (is_update_file) {
505 // Update the file with current parameters
506 SaveParameterToFile(parameter_file_name_, &frame_decoder_parameters_);
507 }
508
509 return DPC_E_OK;
510}

566 int IscFramedecoderInterface::GetDecodeData(IscImageInfo* isc_image_Info, IscBlockDisparityData* isc_block_disparity_data)
567{
568
569 if (isc_image_Info->grab != IscGrabMode::kParallax) {
570 return DPCPROCESS_E_INVALID_MODE;
571 }
572
573 int fd_index = kISCIMAGEINFO_FRAMEDATA_LATEST;
574
575 if ((isc_image_Info->frame_data[fd_index].raw.width == 0) || (isc_image_Info->frame_data[fd_index].raw.height == 0)) {
576 return DPC_E_OK;
577 }
578
579 if (parameter_update_request_) {
580 int ret = SetParameterToFrameDecoderModule(&frame_decoder_parameters_);
581 parameter_update_request_ = false;
582 }
583
584 // (1)
585 int width = isc_image_Info->frame_data[fd_index].raw.width / 2;
586 int height = isc_image_Info->frame_data[fd_index].raw.height;
587 unsigned char* raw_image = isc_image_Info->frame_data[fd_index].raw.image;
588
589 work_buffers_.buff_image[0].width = width;
590 work_buffers_.buff_image[0].height = height;
591 work_buffers_.buff_image[0].channel_count = 1;
592 unsigned char* prgtimg = work_buffers_.buff_image[0].image;
593
594 work_buffers_.buff_image[1].width = width;
595 work_buffers_.buff_image[1].height = height;
596 work_buffers_.buff_image[1].channel_count = 1;
597 unsigned char* plftimg = work_buffers_.buff_image[1].image;
598
599 work_buffers_.buff_image[2].width = width;
600 work_buffers_.buff_image[2].height = height;
601 work_buffers_.buff_image[2].channel_count = 1;
602
603 ISCFrameDecoder::decodeFrameData(
604 height, // 画像の高さ
605 width, // 画像の幅
606 raw_image, // フレームデータ
607 prgtimg, // 右（基準）画像データ 右下原点
608 plftimg); // 左（比較）画像データまたは視差エンコードデータ 右下原点
609
610
611 // (2)
612 int decode_width = isc_image_Info->frame_data[fd_index].raw.width / 2;
613 int decode_height = isc_image_Info->frame_data[fd_index].raw.height;
614
615 work_buffers_.buff_image[3].width = decode_width;
616 work_buffers_.buff_image[3].height = decode_height;
617 work_buffers_.buff_image[3].channel_count = 1;
618 unsigned char* pdspimg = work_buffers_.buff_image[3].image;
619
620 int* pblkhgt = &isc_block_disparity_data->blkhgt;
621 int* pblkwdt = &isc_block_disparity_data->blkwdt;
622 int* pmtchgt = &isc_block_disparity_data->mtchgt;
623 int* pmtcwdt = &isc_block_disparity_data->mtcwdt;
624 int* pblkofsx = &isc_block_disparity_data->dspofsx;
625 int* pblkofsy = &isc_block_disparity_data->dspofsy;
626 int* pdepth = &isc_block_disparity_data->depth;
627 int* pshdwdt = &isc_block_disparity_data->shdwdt;
628
629 float* ppxldsp = isc_block_disparity_data->ppxldsp;
630 float* pblkdsp = isc_block_disparity_data->pblkdsp;
631 int* pblkval = isc_block_disparity_data->pblkval;
632 int* pblkcrst = isc_block_disparity_data->pblkcrst;
633
634 ISCFrameDecoder::getDisparityData(
635 decode_height, // 画像の高さ
636 decode_width, // 画像の幅
637 prgtimg, // 右（基準）画像データ 右下原点
638 plftimg, // 視差エンコードデータ
639 pblkhgt, // 視差ブロック高さ
640 pblkwdt, // 視差ブロック幅
641 pmtchgt, // マッチングブロック高さ
642 pmtcwdt, // マッチングブロック幅
643 pblkofsx, // 視差ブロック横オフセット
644 pblkofsy, // 視差ブロック縦オフセット
645 pdepth, // マッチング探索幅
646 pshdwdt, // 画像遮蔽幅
647 pdspimg, // 視差画像 右下原点
648 ppxldsp, // 視差データ 右下原点
649 pblkdsp, // ブロック視差データ 右下基点
650 pblkval, // 視差ブロック視差値(1000倍サブピクセル精度整数)
651 pblkcrst); // ブロックコントラスト
652
653 isc_block_disparity_data->image_width = isc_image_Info->frame_data[fd_index].p1.width;
654 isc_block_disparity_data->image_height = isc_image_Info->frame_data[fd_index].p1.height;
655
656 return DPC_E_OK;
657}

667int IscFramedecoderInterface::GetDecodeDataDoubleShutter(IscImageInfo* isc_image_info_in, IscBlockDisparityData* isc_block_disparity_data, IscImageInfo* isc_image_info_out)
668{
669
670 if (isc_image_info_in->grab != IscGrabMode::kParallax) {
671 return DPCPROCESS_E_INVALID_MODE;
672 }
673
674 int fd_index = kISCIMAGEINFO_FRAMEDATA_LATEST;
675
676 if ((isc_image_info_in->frame_data[fd_index].raw.width == 0) || (isc_image_info_in->frame_data[fd_index].raw.height == 0)) {
677 return DPC_E_OK;
678 }
679
680 fd_index = kISCIMAGEINFO_FRAMEDATA_PREVIOUS;
681 if ((isc_image_info_in->frame_data[fd_index].raw.width == 0) || (isc_image_info_in->frame_data[fd_index].raw.height == 0)) {
682 return DPC_E_OK;
683 }
684
685 if (parameter_update_request_) {
686 int ret = SetParameterToFrameDecoderModule(&frame_decoder_parameters_);
687 parameter_update_request_ = false;
688 }
689
690 // (1)
691 fd_index = kISCIMAGEINFO_FRAMEDATA_LATEST;
692 int width_latest = isc_image_info_in->frame_data[fd_index].raw.width / 2;
693 int height_latest = isc_image_info_in->frame_data[fd_index].raw.height;
694 unsigned char* raw_image_latest = isc_image_info_in->frame_data[fd_index].raw.image;
695
696 isc_image_info_out->frame_data[fd_index].p1.width = width_latest;
697 isc_image_info_out->frame_data[fd_index].p1.height = height_latest;
698 isc_image_info_out->frame_data[fd_index].p1.channel_count = 1;
699 unsigned char* prgtimg_latest = isc_image_info_out->frame_data[fd_index].p1.image;
