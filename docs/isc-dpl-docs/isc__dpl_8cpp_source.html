94int IscDpl::Initialize(const IscDplConfiguration* ipc_dpl_configuration)
95{
96
97 if (isc_main_control_ != nullptr) {
98 return ISCDPL_E_OPVERLAPED_OPERATION;
99 }
100
101 isc_main_control_ = new IscMainControl;
102 int ret = isc_main_control_->Initialize(ipc_dpl_configuration);
103 if (ret != DPC_E_OK) {
104 return ret;
105 }
106
107 return DPC_E_OK;
108}

116int IscDpl::Terminate()
117{
118
119 if (isc_main_control_ == nullptr) {
120 return ISCDPL_E_INVALID_HANDLE;
121 }
122
123 int ret = isc_main_control_->Terminate();
124 if (ret != DPC_E_OK) {
125 // continue on error!
126 }
127
128 delete isc_main_control_;
129 isc_main_control_ = nullptr;
130
131 return DPC_E_OK;
132}
