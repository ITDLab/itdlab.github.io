90int IscDpl::Initialize(const IscDplConfiguration* ipc_dpl_configuration)
91{
92
93 if (isc_main_control_ != nullptr) {
94 return ISCDPL_E_OPVERLAPED_OPERATION;
95 }
96
97 isc_main_control_ = new IscMainControl;
98 int ret = isc_main_control_->Initialize(ipc_dpl_configuration);
99 if (ret != DPC_E_OK) {
100 return ret;
101 }
102
103 return DPC_E_OK;
104}

112int IscDpl::Terminate()
113{
114
115 if (isc_main_control_ == nullptr) {

126
127 return DPC_E_OK;
