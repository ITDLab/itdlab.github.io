105int IscBlockMatchingInterface::Initialize(IscDataProcModuleConfiguration* isc_data_proc_module_configuration)
106{
107 swprintf_s(isc_data_proc_module_configuration_.configuration_file_path, L"%s", isc_data_proc_module_configuration->configuration_file_path);
108 swprintf_s(isc_data_proc_module_configuration_.log_file_path, L"%s", isc_data_proc_module_configuration->log_file_path);
109
110 isc_data_proc_module_configuration_.log_level = isc_data_proc_module_configuration->log_level;
111 isc_data_proc_module_configuration_.isc_camera_model = isc_data_proc_module_configuration->isc_camera_model;
112 isc_data_proc_module_configuration_.max_image_width = isc_data_proc_module_configuration->max_image_width;
113 isc_data_proc_module_configuration_.max_image_height = isc_data_proc_module_configuration->max_image_height;
114
115 switch (isc_data_proc_module_configuration_.isc_camera_model) {
116 case IscCameraModel::kVM:
117 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_VM.ini", isc_data_proc_module_configuration_.configuration_file_path);
118 break;
119
120 case IscCameraModel::kXC:
121 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_XC.ini", isc_data_proc_module_configuration_.configuration_file_path);
122 break;
123
124 case IscCameraModel::k4K:
125 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_4K.ini", isc_data_proc_module_configuration_.configuration_file_path);
126 break;
127
128 case IscCameraModel::k4KA:
129 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_4KA.ini", isc_data_proc_module_configuration_.configuration_file_path);
130 break;
131
132 case IscCameraModel::k4KJ:
133 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_4KJ.ini", isc_data_proc_module_configuration_.configuration_file_path);
134 break;
135
136 default:
137 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter.ini", isc_data_proc_module_configuration_.configuration_file_path);
138 break;
139 }
140
141 int ret = LoadParameterFromFile(parameter_file_name_, &block_matching_parameters_);
142 if (ret != DPC_E_OK) {
143 if (ret == DPCPROCESS_E_FILE_NOT_FOUND) {
144 // Since the file does not exist, it will be created by default and continue
145 SaveParameterToFile(parameter_file_name_, &block_matching_parameters_);
146 }
147 else {
148 return ret;
149 }
150 }
151
152 block_matching_parameters_.matching_parameter.imghgt = isc_data_proc_module_configuration_.max_image_height;
153 block_matching_parameters_.matching_parameter.imgwdt = isc_data_proc_module_configuration_.max_image_width;
154
155 ret = SetParameterToBlockMatchingModule(&block_matching_parameters_);
156 if (ret != DPC_E_OK) {
157 return ret;
158 }
159
160 // get work
161 size_t image_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
162 size_t depth_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
163
164 for (int i = 0; i < 2; i++) {
165 work_buffers_.buff_image[i].width = 0;
166 work_buffers_.buff_image[i].height = 0;
167 work_buffers_.buff_image[i].channel_count = 0;
168 work_buffers_.buff_image[i].image = new unsigned char[image_size];
169 memset(work_buffers_.buff_image[i].image, 0, image_size);
170 }
171
172 // initialize FrameDecoder
173 BlockMatching::initialize(isc_data_proc_module_configuration_.max_image_height, isc_data_proc_module_configuration_.max_image_width);
174 BlockMatching::createMatchingThread();
175
176 // Check if OpenCL is available. Enable it if it is available.
177 if (block_matching_parameters_.system_parameter.enabled_opencl_for_avedisp && cv::ocl::haveOpenCL()) {
178 // it can use openCL
179 //cv::String build_info_str = cv::getBuildInformation();
180 //OutputDebugStringA(build_info_str.c_str());
181 BlockMatching::setUseOpenCLForBlockMatching(1);
182 }
183 else {
184 BlockMatching::setUseOpenCLForBlockMatching(0);
185 }
186
187 return DPC_E_OK;
188}

750{

760
761 if ((isc_image_Info->frame_data[fd_index].p2.width == 0) || (isc_image_Info->frame_data[fd_index].p2.height == 0)) {

766 int ret = SetParameterToBlockMatchingModule(&block_matching_parameters_);

771
772 // (1) matching

774 unsigned char* s1_image = isc_image_Info->frame_data[fd_index].p2.image;

777
778 // (2) get disparity

791 return DPC_E_OK;
