105int IscBlockMatchingInterface::Initialize(IscDataProcModuleConfiguration* isc_data_proc_module_configuration)
106{
107 swprintf_s(isc_data_proc_module_configuration_.configuration_file_path, L"%s", isc_data_proc_module_configuration->configuration_file_path);
108 swprintf_s(isc_data_proc_module_configuration_.log_file_path, L"%s", isc_data_proc_module_configuration->log_file_path);
109
110 isc_data_proc_module_configuration_.log_level = isc_data_proc_module_configuration->log_level;
111 isc_data_proc_module_configuration_.isc_camera_model = isc_data_proc_module_configuration->isc_camera_model;
112 isc_data_proc_module_configuration_.max_image_width = isc_data_proc_module_configuration->max_image_width;
113 isc_data_proc_module_configuration_.max_image_height = isc_data_proc_module_configuration->max_image_height;
114
115 switch (isc_data_proc_module_configuration_.isc_camera_model) {
116 case IscCameraModel::kVM:
117 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_VM.ini", isc_data_proc_module_configuration_.configuration_file_path);
118 break;
119
120 case IscCameraModel::kXC:
121 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_XC.ini", isc_data_proc_module_configuration_.configuration_file_path);
122 break;
123
124 case IscCameraModel::k4K:
125 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_4K.ini", isc_data_proc_module_configuration_.configuration_file_path);
126 break;
127
128 case IscCameraModel::k4KA:
129 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_4KA.ini", isc_data_proc_module_configuration_.configuration_file_path);
130 break;
131
132 case IscCameraModel::k4KJ:
133 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter_4KJ.ini", isc_data_proc_module_configuration_.configuration_file_path);
134 break;
135
136 default:
137 swprintf_s(parameter_file_name_, L"%s\\BlockMatcingParameter.ini", isc_data_proc_module_configuration_.configuration_file_path);
138 break;
139 }
140
141 int ret = LoadParameterFromFile(parameter_file_name_, &block_matching_parameters_);
142 if (ret != DPC_E_OK) {
143 if (ret == DPCPROCESS_E_FILE_NOT_FOUND) {
144 // Since the file does not exist, it will be created by default and continue
145 SaveParameterToFile(parameter_file_name_, &block_matching_parameters_);
146 }
147 else {
148 return ret;
149 }
150 }
151
152 block_matching_parameters_.matching_parameter.imghgt = isc_data_proc_module_configuration_.max_image_height;
153 block_matching_parameters_.matching_parameter.imgwdt = isc_data_proc_module_configuration_.max_image_width;
154
155 ret = SetParameterToBlockMatchingModule(&block_matching_parameters_);
156 if (ret != DPC_E_OK) {
157 return ret;
158 }
159
160 // get work
161 size_t image_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
162 size_t depth_size = isc_data_proc_module_configuration_.max_image_width * isc_data_proc_module_configuration_.max_image_height;
163
164 for (int i = 0; i < 4; i++) {
165 work_buffers_.buff_image[i].width = 0;
166 work_buffers_.buff_image[i].height = 0;
167 work_buffers_.buff_image[i].channel_count = 0;
168 work_buffers_.buff_image[i].image = new unsigned char[image_size];
169 memset(work_buffers_.buff_image[i].image, 0, image_size);
170
171 work_buffers_.buff_depth[i].width = 0;
172 work_buffers_.buff_depth[i].height = 0;
173 work_buffers_.buff_depth[i].image = new float[depth_size];
174 memset(work_buffers_.buff_depth[i].image, 0, depth_size * sizeof(float));
175 }
176
177 // initialize FrameDecoder
178 BlockMatching::initialize(isc_data_proc_module_configuration_.max_image_height, isc_data_proc_module_configuration_.max_image_width);
179 BlockMatching::createMatchingThread();
180
181 // Check if OpenCL is available. Enable it if it is available.
182 if (block_matching_parameters_.system_parameter.enabled_opencl_for_avedisp && cv::ocl::haveOpenCL()) {
183 // it can use openCL
184 //cv::String build_info_str = cv::getBuildInformation();
185 //OutputDebugStringA(build_info_str.c_str());
186 BlockMatching::setUseOpenCLForBlockMatching(1);
187 }
188 else {
189 BlockMatching::setUseOpenCLForBlockMatching(0);
190 }
191
192 return DPC_E_OK;
193}

374{
375 BlockMatching::deleteMatchingThread();
376
377 // release work
378 for (int i = 0; i < 4; i++) {
379 work_buffers_.buff_image[i].width = 0;
380 work_buffers_.buff_image[i].height = 0;
381 work_buffers_.buff_image[i].channel_count = 0;
382 delete[] work_buffers_.buff_image[i].image;
383 work_buffers_.buff_image[i].image = nullptr;
384
385 work_buffers_.buff_depth[i].width = 0;
386 work_buffers_.buff_depth[i].height = 0;
387 delete[] work_buffers_.buff_depth[i].image;
388 work_buffers_.buff_depth[i].image = nullptr;
389 }
390
391 return DPC_E_OK;
392

760{
761 if (isc_image_Info->grab != IscGrabMode::kCorrect) {
762 return DPCPROCESS_E_INVALID_MODE;
763 }
764
765 int fd_index = kISCIMAGEINFO_FRAMEDATA_LATEST;
766
767 if ((isc_image_Info->frame_data[fd_index].p1.width == 0) || (isc_image_Info->frame_data[fd_index].p1.height == 0)) {
768 return DPC_E_OK;
769 }
770
771 if ((isc_image_Info->frame_data[fd_index].p2.width == 0) || (isc_image_Info->frame_data[fd_index].p2.height == 0)) {
772 return DPC_E_OK;
773 }
774
775 if (parameter_update_request_) {
776 int ret = SetParameterToBlockMatchingModule(&block_matching_parameters_);
777 parameter_update_request_ = false;
778 }
779
780 IscImageInfo* dst_isc_image_info = &isc_data_proc_result_data->isc_image_info;
781
782 // (1) matching
783 unsigned char* bade_image = isc_image_Info->frame_data[fd_index].p1.image;
784 unsigned char* compare_image = isc_image_Info->frame_data[fd_index].p2.image;
785
786 BlockMatching::matching(bade_image, compare_image);
787
788 // (2) get disparity
789
790 int width = isc_image_Info->frame_data[fd_index].p1.width;
791 int height = isc_image_Info->frame_data[fd_index].p1.height;
792
793 unsigned char* display_image = work_buffers_.buff_image[0].image;
794
795 dst_isc_image_info->frame_data[fd_index].depth.width = width;
796 dst_isc_image_info->frame_data[fd_index].depth.height = height;
797 float* disparity = dst_isc_image_info->frame_data[fd_index].depth.image;
798
799 BlockMatching::getDisparity(height, width, display_image, disparity);
800
801 return DPC_E_OK;
802}
