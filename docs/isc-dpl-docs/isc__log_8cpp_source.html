76void IscLog::Open(wchar_t* save_path, wchar_t* save_file_name, int log_level, bool is_immediate_mode)
77{
78
79 // init
80 thread_terminate_done_ = 0;
81 thread_terminate_request_ = 0;
82 thread_end_code_ = 0;
83 abort_request_ = false;
84
85 log_level_ = log_level;
86 max_file_size_ = 10 * 1024 * 1024;
87 immediate_mode_ = is_immediate_mode;
88
89 if (log_level_ == 0) {
90 // 保存しない
91 return;
92 }
93
94 // folder check
95 swprintf_s(save_path_, L"%s", save_path);
96 swprintf_s(save_file_name_, L"%s", save_file_name);
97
98 if (::PathFileExists(save_path_) && !::PathIsDirectory(save_path_)) {
99 // 指定されたパスにファイルが存在、かつディレクトリでない
100 }
101 else if (::PathFileExists(save_path_)) {
102 // 指定されたパスがディレクトリである
103 }
104 else {
105 // 作成する
106 SHCreateDirectoryEx(NULL, save_path_, NULL);
107 }
108
109 // open file
110 if (file_opened_) {
111 fclose(file_);
112 file_ = NULL;
113 file_opened_ = false;
114 }
115
116 SYSTEMTIME st;
117 GetLocalTime(&st);
118 swprintf_s(log_file_name_, L"%s-%02d%02d%02d%02d%02d.txt", save_file_name_, st.wMonth, st.wDay, st.wHour, st.wMinute, st.wSecond);
119
120 // Threadで書き込みます
121 if (immediate_mode_) {
122 // 書き込み時にファイルを開き直します
123 errno_t err = _wfopen_s(&file_, log_file_name_, L"wt+, ccs=UTF-8");
124 if (file_ == NULL) {
125 file_opened_ = false;
126 }
127 else {
128 fclose(file_);
129 file_ = NULL;
130 file_opened_ = false;
131 }
132 }
133 else {
134 errno_t err = _wfopen_s(&file_, log_file_name_, L"wt+, ccs=UTF-8");
135 if (file_ == NULL) {
136 file_opened_ = false;
137 }
138 else {
139 file_opened_ = true;
140 }
141 }
142
143 log_data_que_ = new LogDataQue;
144 log_data_que_->open(128, false); // mode=all data
145
146 // thread start
147 wchar_t semaphore_name[64] = {};
148 swprintf_s(semaphore_name, sizeof(semaphore_name) / sizeof(wchar_t), L"MAIN_THREAD_SEMAPHORENAME_%d", 0);
149 handle_semaphore_ = CreateSemaphore(NULL, 0, 1, semaphore_name);
150 if (handle_semaphore_ == NULL) {
151 // Fail
152 }
153 InitializeCriticalSection(&threads_critical_);
154
155 if ((thread_handle_ = (HANDLE)_beginthreadex(0, 0, LogThread, (void*)this, 0, 0)) == 0) {
156 // Fail
157 }
158
159 // THREAD_PRIORITY_TIME_CRITICAL
160 // THREAD_PRIORITY_HIGHEST +2
161 // THREAD_PRIORITY_ABOVE_NORMAL +1
162 // THREAD_PRIORITY_NORMAL +0
163 // THREAD_PRIORITY_BELOW_NORMAL -1
164 if (thread_handle_ != NULL) {
165 SetThreadPriority(thread_handle_, THREAD_PRIORITY_BELOW_NORMAL);
166 }
167
168 return;
169}

177void IscLog::Close()
178{
179
180 // stop thread and release
181 //
182 // 1回はLoopさせておく
183 Sleep(60);
184
185 thread_terminate_done_ = 0;
186 thread_terminate_request_ = 1;
187 thread_end_code_ = 0;
188
189 ReleaseSemaphore(handle_semaphore_, 1, NULL);
190 Sleep(60);
191
192 int count = 0;
193 while (thread_terminate_done_ == 0) {
194 if (count > 100) {
195 break;
196 }
197 count++;
198 Sleep(10L);
199 }
200
201 if (thread_terminate_done_ == 0) {
202 abort_request_ = true;
203
204 count = 0;
205 while (thread_terminate_done_ == 0) {
206 if (count > 100) {
207 break;
208 }
209 count++;
210 Sleep(10L);
211 }
212 }
213
214 if (thread_handle_ != NULL) {
215 CloseHandle(thread_handle_);
216 thread_handle_ = NULL;
217 }
218 if (handle_semaphore_ != NULL) {
219 CloseHandle(handle_semaphore_);
220 handle_semaphore_ = NULL;
221 }
222 DeleteCriticalSection(&threads_critical_);
223
224 Flush();
225
226 if (file_ && file_opened_) {
227 fclose(file_);
228 file_ = NULL;
229 file_opened_ = false;
230 }
231
232 return;
233}

260void IscLog::LogForThread(const SYSTEMTIME* st, const wchar_t *head, Level level, const wchar_t *str)
261{
262 if (log_level_ == 0) {
263 // off
264 return;
265 }
266
267 if (immediate_mode_) {
268 errno_t err = _wfopen_s(&file_, log_file_name_, L"at+, ccs=UTF-8");
269 if (file_ == NULL) {
270 file_opened_ = false;
271 }
272 else {
273 file_opened_ = true;
274 }
275 }
276
277 if (file_ && file_opened_) {
278
279 bool en_log = false;
280 if (log_level_ == 1) {
281 // エラーを保存(Debug含む）
282 if ((int)level >= (int)Level::Error) {
283 en_log = true;
284 }
285 }
286 else if (log_level_ == 2) {
287 // ワーニングを保存
288 if ((int)level >= (int)Level::Warning) {
289 en_log = true;
290 }
291 }
292 else if (log_level_ >= 3) {
293 // 全て保存
294 if ((int)level >= (int)Level::Information) {
295 en_log = true;
296 }
297 }
298
299 if (en_log) {
300
301 const wchar_t *type;
302 switch (level)
303 {
304 case Level::Debug:
305 type = L"DEBUG";
306 break;
307 case Level::Error:
308 type = L"ERROR";
309 break;
310 case Level::Warning:
311 type = L"WARN";
312 break;
313 case Level::Information:
314 type = L"INFO";
315 break;
316 default:
317 type = L"UNKOWN";
318 break;
319 }
320
321 wchar_t dt[1024] = {};
322 swprintf_s(dt, L"%02d%02d:%02d%02d%02d.%03d[%s][%s] ", st->wMonth, st->wDay, st->wHour, st->wMinute, st->wSecond, st->wMilliseconds, head, type);
323
324 fputws(dt, file_);
325
326 file_size_ += (int)wcslen(dt);
327 fputws(str, file_);
328
329 file_size_ += (int)wcslen(str);
330
331 if (level == Level::Error) {
332 fflush(file_);
333 }
334
335 if (file_size_ >= max_file_size_) {
336 fclose(file_);
337 swprintf_s(log_file_name_, L"%s-%02d%02d%02d%02d%02d.txt", save_file_name_, st->wMonth, st->wDay, st->wHour, st->wMinute, st->wSecond);
338
339 if (immediate_mode_) {
340 // 書き込み時にファイルを開き直します
341 errno_t err = _wfopen_s(&file_, log_file_name_, L"wt+, ccs=UTF-8");
342 if (file_ == NULL) {
343 file_opened_ = false;
344 }
345 else {
346 file_opened_ = true;
347 }
348 }
349 else {
350 errno_t err = _wfopen_s(&file_, log_file_name_, L"wt+, ccs=UTF-8");
351 if (file_ == NULL) {
352 file_opened_ = false;
353 }
354 else {
355 file_opened_ = true;
356 }
357 }
358
359 file_size_ = 0;
360 }
361 }
362 }
363
364 if (immediate_mode_ && (file_ != NULL)) {
365 Flush();
366 fclose(file_);
367 file_ = NULL;
368 file_opened_ = false;
369 }
370
371 return;
372}

487int IscLog::LogProc()
488{
489
490 DWORD sleep_time = 30;
491
492 wchar_t head[2048] = {};
493 wchar_t buffer[2048] = {};
494 int count = 0;
495 Level level = Level::Information;
496 SYSTEMTIME st;
497
498 while (thread_terminate_request_ < 1) {
499
500 int get_index = log_data_que_->get(&st, &level, head, sizeof(head) / sizeof(wchar_t), buffer, sizeof(buffer) / sizeof(wchar_t));
501 if (get_index >= 0) {
502
503 // Log
504 switch (level)
505 {
506 case Level::Debug:
507 this->LogForThread(&st, head, level, buffer);
508 break;
509 case Level::Error:
510 this->LogForThread(&st, head, level, buffer);
511 break;
512 case Level::Warning:
513 this->LogForThread(&st, head, level, buffer);
514 break;
515 case Level::Information:
516 this->LogForThread(&st, head, level, buffer);
517 break;
518 default:
519 break;
520 }
521
522 log_data_que_->done(get_index);
523
524 // 一旦全部書き込む
525 while (true) {
526
527 if (abort_request_) {
528 break;
529 }
530
531 int get_index = log_data_que_->get(&st, &level, head, sizeof(head) / sizeof(wchar_t), buffer, sizeof(buffer) / sizeof(wchar_t));
532 if (get_index >= 0) {
533
534 // Log
535 switch (level)
536 {
537 case Level::Debug:
538 this->LogForThread(&st, head, level, buffer);
539 break;
540 case Level::Error:
541 this->LogForThread(&st, head, level, buffer);
542 break;
543 case Level::Warning:
544 this->LogForThread(&st, head, level, buffer);
545 break;
546 case Level::Information:
547 this->LogForThread(&st, head, level, buffer);
548 break;
549 default:
550 break;
551 }
552
553 log_data_que_->done(get_index);
554 }
555 else {
556 break;
557 }
558 }
559
560 sleep_time = 1;
561 }
562 else {
563 sleep_time = 30;
564 }
565
566 Sleep(sleep_time);
567 }
568
569 thread_terminate_request_ = 0;
570 thread_terminate_done_ = 1;
571
572 return 0;
573}
